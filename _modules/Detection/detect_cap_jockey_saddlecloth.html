[docs]def detect_cap_jockey_saddlecloth(horseracingResult):

  '''
  A detection model for detecting cap bounding boxes, jockey bounding boxes, jockey masks and saddlecloth bounding boxes. All in 1 model
  
  Parameters
  ----------
  horseracingResult : object
  
  Notes
  -----
  The detection model is maintained by Suzy
  '''
  print('detect cap jockey and saddlecloth ... ')
  vid_name = horseracingResult.processing_vid_name
  print(vid_name)
  track = horseracingResult.race_info[vid_name]["track"]
  print(track)
  model_dir = horseracingResult.model_dir
  result_dir = horseracingResult.result_dir

  start_frame, end_frame = horseracingResult.start_end_frames[vid_name]
  print('Starting Frame is', start_frame, 'and Ending Frame is', end_frame)
  if start_frame >= end_frame or start_frame < 0 or end_frame < 0:
    print('Start/End frame error, end of detection')
    return

  config_file, _, _ = config.get_model_info(model_dir, track, "Detection")
  # update the config options with the config file
  cfg.merge_from_file(config_file)
  cfg['MODEL']['WEIGHT'] = os.path.join(os.path.dirname(config_file), cfg['MODEL']['WEIGHT']) # change the weight path to include the folder path
  
  maskscoring = MaskScoringRCNN(cfg,min_image_size=cfg['INPUT']['MIN_SIZE_TEST'],confidence_threshold=0.9)
  print("Network Loaded (Detection)")

  horseracingResult.load_frames(vid_name, start_frame, end_frame)

  detected_caps, detected_saddlecloths, detected_jockeys = {}, {}, {}

  all_masks, all_labels = [], []

  ti = time.time()

  for frm in range(len(horseracingResult.frames)):
    
    if frm % 100 == 0: # print progress per 100 frames processed
      print('frame ', frm, '/', len(horseracingResult.frames))
    
    image = decode_img(horseracingResult.frames[frm])
      
    boxes, masks, labels = maskscoring.get_result_by_frame(image)
    labels = labels.numpy().tolist()
  
    caps, saddlecloths, jockeys = [], [], []
    
    for lid, l in enumerate(labels):
      
      box = boxes[lid][:4]
      score = boxes[lid][4]
      mask = masks[lid,0,:,:]

      if l == 2:
        caps.append({"bbox": box, "score": float(score)})
      elif l == 1:
        mask = mask[box[1]:box[3],box[0]:box[2]]
        mask = np.asfortranarray(np.asarray(mask, np.uint8))
        m_string = bytes.decode(mask_save_tool.encode(mask)["counts"]) # convert the mask to a string for ease of storage
        jockeys.append({"bbox": box, "score": float(score), "rle_string": m_string})
      elif l == 3:
        saddlecloths.append({"bbox": box, "score": float(score)})

    detected_caps["frame_" + str(frm + start_frame)] = caps
    detected_saddlecloths["frame_" + str(frm + start_frame)] = saddlecloths
    detected_jockeys["frame_" + str(frm + start_frame)] = jockeys

  # write to json
  with open(os.path.join(horseracingResult.result_dir, config.directories[config.Task.DETECTION]['ROOT'], vid_name + ".json"), 'w') as file:
    json.dump({"target_frame_size": [ 1920, 1080 ], "start_frame": int(start_frame), "finish_frame": int(end_frame),
           'cap': detected_caps, 'jockey': detected_jockeys, 'saddlecloth': detected_saddlecloths}, file)
    file.flush()
    file.close()

  print("Detection Feature Saved: " + str(time.time()-ti))
