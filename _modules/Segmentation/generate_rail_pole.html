from detectron2 import model_zoo
from detectron2.engine import DefaultPredictor
from detectron2.config import get_cfg

import os, cv2, random, json
import numpy as np
import config

from pycocotools import mask as mask_save_tool


[docs]def generate_rail_pole(horseracingResult):
    
    '''
    generate rail pole
    This model is trained with detectron2. Therefore, it only requires a few lines of code to load, and test the model.
    '''
    
    print("Generating Rail Masks")
    vid_name = horseracingResult.processing_vid_name
    track = horseracingResult.race_info[vid_name]["track"] 
    network_path, mean_path, batch = config.get_model_info(horseracingResult.model_dir, track, "RailPoleSegmentation")
        
    start_frame, end_frame = horseracingResult.start_end_frames[vid_name]
    horseracingResult.load_frames(vid_name, start_frame, end_frame)
    
    # assuming data only exists in terms of framesetids but not racelabels
    cfg = get_cfg()
    cfg.merge_from_file(model_zoo.get_config_file("COCO-InstanceSegmentation/mask_rcnn_R_101_FPN_3x.yaml"))
    cfg.DATALOADER.NUM_WORKERS = 2
    cfg.MODEL.ROI_HEADS.NUM_CLASSES = 1  # only has one class (ballon).
    cfg.MODEL.WEIGHTS = network_path  # path to the model we just trained
    cfg.MODEL.ROI_HEADS.SCORE_THRESH_TEST = 0.7 # set a custom testing threshold

    predictor = DefaultPredictor(cfg)
    
    rail_mask_prediction = {"target_frame_size": [ 1920, 1080 ],"data": {"start_frame": int(start_frame), "finish_frame": int(end_frame)}}
    
    for frm in range(horseracingResult.frames.shape[0]):
      outputs = predictor(horseracingResult.frames[frm])
      
      boxes = outputs["instances"].to("cpu").pred_boxes.tensor.numpy()
      scores = outputs["instances"].to("cpu").scores.numpy()
      masks = outputs["instances"].to("cpu").pred_masks.numpy()
      
      output = []
      for b in range(boxes.shape[0]):
        if scores[b] > 0.5:
          x0, y0, x1, y1 = boxes[b]
          x0, y0, x1, y1 = int(x0), int(y0), int(x1), int(y1)
          width, height = int(x1 - x0), int(y1 - y0)
          
          mask = masks[b,x0:x1,y0:y1]
          mask[mask>0.8] = 1
          mask[mask<=0.8] = 0
          # convert to string for ease of saving
          mask = np.asfortranarray(np.asarray(mask, np.uint8))
          m_string = bytes.decode(mask_save_tool.encode(mask)["counts"])
         
          output.append({"bbox": [x0, y0, width, height],
                     "score": float(scores[b]),
                     "rle_string": m_string})

      rail_mask_prediction["data"]["frame_" + str(frm + start_frame)] = output
      if frm%100 == 0:
        print("Frame " + str(frm) + " of " + str(horseracingResult.frames.shape[0]))
    
    # release memory
    del predictor
              
    # write to json
    with open(os.path.join(horseracingResult.result_dir, "Rail_Pole_Masks", vid_name + ".json"), 'w') as file:
      json.dump(rail_mask_prediction, file)
      file.flush()
      file.close()
      
    print("Finish writing rail poles")
