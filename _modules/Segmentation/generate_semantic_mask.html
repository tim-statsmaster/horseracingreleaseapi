import json, os, time
import config
import numpy as np

from pycocotools import mask as mask_save_tool
from multiprocessing.managers import BaseManager

'''
BaseManger is used for launching a separate process to host the tensorflow model.
This is very important because the tensorflow code will take up GPU memory that cannot be released after execution.
By launching a separate process for tensorflow and kill it after execution can solve this problem.
'''


[docs]class Segmentation:

  '''
  This class is used for segmentation of the frame into 7 classes ["background", "race_track", "jockey_horse", "foreground", "pole", "rail", "static"]
  race_track refers to the grass track
  jockey_horse refers to the combination of jockey bodies and horses
  pole refers to the light poles in the field
  static refers to the additional display overlay on the frame by the tv broadcast
  '''

  def __init__(self, network_path):
  
      from Segmentation.Model import Model      
      self.model = Model(network_path, channels=3, n_class=7, img_rows=1080, img_cols=1920, cropping=0)


[docs]  def get_mask(self, frame):
    frame = frame / 255.0 # preprocess image 
    return self.model.predict(frame)

    

[docs]def generate_semantic_mask(horseracingResult):
    
    print("Generating Semantic Masks")
    vid_name = horseracingResult.processing_vid_name 
    track = horseracingResult.race_info[vid_name]["track"]
    network_path, _, batch = config.get_model_info(horseracingResult.model_dir, track, "SemanticSegmentation")
    
    BaseManager.register('Segmentation', Segmentation)
    manager = BaseManager()
    manager.start()
    segmentation_manager = manager.Segmentation(network_path)
    
    start_frame, end_frame = horseracingResult.start_end_frames[vid_name]
    horseracingResult.load_frames(vid_name, start_frame, end_frame)
    
    rail_mask_prediction = {"object_id_list":[0,1,2,3,4,5,6], "object_name_list": ["background", "race_track", "jockey_horse", "foreground", "pole", "rail", "static"], "target_frame_size": [ 1920, 1080 ],
    "data": {"start_frame": int(start_frame), "finish_frame": int(end_frame)}}
    
    for frm in range(horseracingResult.frames.shape[0]):
      masks = segmentation_manager.get_mask(horseracingResult.frames[frm])
      
      output = []    
      for mid in range(7):
        m = masks == mid
        # convert the mask into string for ease of storage
        m = np.asfortranarray(m)
        m_string = bytes.decode(mask_save_tool.encode(m)["counts"])
      
        output.append({"object_id": mid, "rle_string": m_string})
                       
      rail_mask_prediction["data"]["frame_" + str(frm + start_frame)] = output
      if frm%100 == 0:
        print("Frame " + str(frm) + " of " + str(horseracingResult.frames.shape[0]))
    
    # release memory
    manager.shutdown()
              
    # write to json
    with open(os.path.join(horseracingResult.result_dir, "Semantic_Masks", vid_name + ".json"), 'w') as file:
      json.dump(rail_mask_prediction, file)
      file.flush()
      file.close()
    print("Finish writing segmentation masks")
