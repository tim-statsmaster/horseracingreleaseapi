[docs]def match_with_triplet_saddlecloth(subsequences, subseq_ids_scores, triplet_features_caps, triplet_features_jockeys, jockey_cap_relation, detected_boxes, camchange, jockeys, track):

  '''
  This function performs the following functions
  1) divides the subsequences into different subraces
  2) compute the mean triplet cap and jockey features for each subsequences
  3) use the triplet thresholds, triplet features, saddlecloth scores, saddlecloth thresholds to pair up the subsequences
  4) the paired up subsequences are separated into batches "matchlist_step" in terms of priority
  5) using the pair of subsequences to form group of subsequences that belongs to the same jockeys
  6) Join the subsequences within the group to form one trajectory
  '''
  
  print("match_with_triplet_saddlecloth ... ")
  ti = time.time()
  sigma = 0.35

  num_subrace = len(camchange) + 1 # subrace means the portion of the race without camera change
  subraces = [[] for i in range (0, num_subrace)]
  
  # divides the subsequences into different subraces
  if len(camchange) == 0:
    subraces[0] = [id for id, sq in enumerate(subsequences)]
  else:
    subraces[0] = [id for id, sq in enumerate(subsequences) if sq[0,-1] <= camchange[0]]
  if num_subrace > 1:
    subraces[-1] = [id for id, sq in enumerate(subsequences) if sq[0,0] > camchange[-1]]
    for i in range(1, num_subrace-1):
      subraces[i] = [id for id, sq in enumerate(subsequences) if sq[0,0] > camchange[i-1] and sq[0,-1] <= camchange[i]]
  

  # get the mean triplet feature for cap and jockey for each subsequences
  triplet_mean_seq_jockey = [ get_sequence_tripfeat(sq, triplet_features_jockeys, jockey_cap_relation=jockey_cap_relation) for sq in subsequences]
  triplet_mean_seq_cap = [ get_sequence_tripfeat(sq, triplet_features_caps, jockey_cap_relation=None) for sq in subsequences]

  match_thres = config.tracking['match_thres']
  triplet_thres = config.tracking['triplet_thres']
  triplet_thres_high = config.tracking['triplet_thres_high']
  triplet_thres_low = config.tracking['triplet_thres_low']
  triplet_thres_jockey = config.tracking['triplet_thres_jockey']
  overlap_thres = config.tracking['overlap_thres']
 
  # separate the subsequences into different batches for matching
  matchlist_steps = [[],[],[],[],[],[],[]]
  
  for feid, sq1 in enumerate(subsequences):
    for beid, sq2 in enumerate(subsequences):
      if feid == beid:
        continue
      
      diff = int(subsequences[beid][0,0] - subsequences[feid][0,subsequences[feid].shape[1]-1])
      if diff < -overlap_thres: #ovelap
        continue

      # calculate cap feature dist and use the dist to calculate cap score
      tripdistavg = np.sum((triplet_mean_seq_cap[feid] - triplet_mean_seq_cap[beid])**2)**0.5
      score_cap = math.exp(-tripdistavg)
 
      # calculate jockey body score and if there is no jockey body for calculation, assign 0.55 to the jockey score.
      if triplet_mean_seq_jockey[feid] != [] and triplet_mean_seq_jockey[beid] != []:
        jockey_tripdistavg =  np.sum((triplet_mean_seq_jockey[feid] - triplet_mean_seq_jockey[beid])**2)**0.5
        score_jockey = math.exp(-jockey_tripdistavg)
      else:
        score_jockey = triplet_thres + 0.5	#add 0.5 to pass step 4 5

      score_triplet = score_cap + score_jockey
      
      st_score = 1.0
      same_subrace = False
      for subrace in subraces:
        if feid in subrace and beid in subrace:
          same_subrace = True
          # if in same subrace, check spatial temporal consistency get st_score
          st_score = check_jump_seqs(feid, beid, subsequences, detected_boxes, overlap_thres)
          score_triplet *= st_score

    # saddlecloth number similary consider predicted id, as well as its score (cumulative probability in a subsequence)
    # logistic sigmoid function f(x) = 1/(1+exp(-sigma*x))
	  # x negative when different number, f(-x) = 1 - f(x)
	  # if one of subsequences has no number, score = 0.5
	  # if both of subsequences has number, they are different, score < 0.5
	  # if both of subsequences has number, they are same, score > 0.5
     
      id1, score1 = subseq_ids_scores[feid]
      id2, score2 = subseq_ids_scores[beid]

      dif_num = True if id1 != id2 and id1 != 0 and id2 != 0 else False
      same_num = True if id1 == id2 and id1 != 0 and id2 != 0 else False
 
      score_scn = 1.0
      score_scn_1 = 1 / (1 + math.exp(-sigma * score1))
      score_scn_2 = 1 / (1 + math.exp(-sigma * score2))
      if dif_num:
        score_scn_1 = 1 - score_scn_1
        score_scn_2 = 1 - score_scn_2
      score_scn = score_scn_1 * score_scn_2
      score_all = score_triplet * score_scn

      #step 0 2 4 for pairs of subsequences in same subraces prior to
      #step 1 3 5 for pairs of subsequences in different subraces
      
      #step 0 1 scoring subsequences with similar cap, similar body and same number 
      if score_all > match_thres and score_jockey >= triplet_thres_jockey:
        if same_subrace:
          matchlist_steps[0].append([feid, beid, score_triplet, score_all])
        else:
          matchlist_steps[1].append([feid, beid, score_triplet, score_all])

      #step 2 3 scoring pairs of subsequences only with similar cap, similar body, without number, high threshold
      elif score_cap > triplet_thres_high or score_jockey > triplet_thres_high:
        if same_subrace:
          matchlist_steps[2].append([feid, beid, score_triplet])
        else:
          matchlist_steps[3].append([feid, beid, score_triplet])

      #step 4 5 same as step 2 3 with normal thresholds
      elif score_cap > triplet_thres or score_jockey > triplet_thres:
        if same_subrace:
          matchlist_steps[4].append([feid, beid, score_triplet])
        else:
          matchlist_steps[5].append([feid, beid, score_triplet])

      #step 6 finaly if two subsequences have same number, accept with lower threshold
      elif same_num and score_cap > triplet_thres_low and score_jockey > triplet_thres_low:
          matchlist_steps[6].append([feid, beid, score_triplet])

  matchlist = [ item for match_step in matchlist_steps for item in sort_list_by_col(match_step)]
  steps = [len(match_step) for match_step in matchlist_steps]
  print('tracking in 7 steps: ', steps)
  
  # form groups with the subsequences and assign them the corresponding ids
  groups, id_groups = search_match_with_id(matchlist, subsequences, subseq_ids_scores, detected_boxes, camchange, steps, triplet_mean_seq_cap, jockeys, track)
  print('num of sequences ', len([[m[0] for m in g] for g in groups]))

  matched_subsequences = []
  # put the sequences together (join subsequences for each group)
  for g in groups:
    sq = np.empty((2,0), dtype=np.int16)
    for m in g:
      # merge subsequence
      overlap_set = set(sq[0,:]).intersection(set(subsequences[int(m[0])][0,:]))
      overlap = len(overlap_set)
      if overlap > 0:
        sq = np.append(sq[:,:sq.shape[1]-overlap], subsequences[int(m[0])][:,overlap:], 1)
      else:  
        sq = np.append(sq, subsequences[int(m[0])][:,overlap:], 1)
    
    matched_subsequences.append(sq)

  # remove too short subsequences and sort them according to group number
  subsequences = matched_subsequences
  subsequences, id_groups  = remove_short_subsequences2(subsequences, id_groups, len_thres)
  subsequences, id_groups  = sort_subsequences2(subsequences, id_groups)
  
  #remove none
  for iid, i in enumerate(id_groups):
    if i == None:
      id_groups[iid] = [0,0]

  print('matching time: ', time.time()-ti)
  
  return subsequences, id_groups

  
  

[docs]def search_match_with_id(mlist, subsequences, subseq_ids_scores, detected_boxes, camchange, steps, triplet_mean_seq, jockeys, track):
        
    match_thres = config.tracking['match_thres']
    triplet_thres = config.tracking['triplet_thres']
    triplet_thres_gp = config.tracking['triplet_thres_gp']
    overlap_thres = config.tracking['overlap_thres']

    t_step_2 = steps[0] + steps[1]
    groups = []
    count_mlist = 0
    matched = set()
    sigma = 0.35
    while len(mlist) != 0:
      s1id, s2id = mlist[0][0], mlist[0][1]
      matched.update([s1id, s2id])
      m1, m2 = [s1id, subsequences[s1id][0,0], subsequences[s1id][0,-1]], [s2id, subsequences[s2id][0,0], subsequences[s2id][0,-1]]
      m1ing = -1
      m2ing = -1
      for gid, g in enumerate(groups):
        if m1 in g:
          m1ing = gid
        if m2 in g:
          m2ing = gid
      
      # if they are already in the same group, no need of merging
      if m1ing == m2ing and m1ing != -1:
        mlist.pop(0)
        count_mlist += 1
        continue
      
      # if it is not belong to any of group, creat a new group

      if m1ing == -1:
        groups.append([m1])
        m1ing = len(groups)-1
      if m2ing == -1:
        groups.append([m2])
        m2ing = len(groups)-1

      # check if the two subsequences have overlapping
      if check_overlap_gps(groups[m1ing], groups[m2ing], subsequences, detected_boxes, overlap_thres) and \
         check_jump_gps(groups[m1ing], groups[m2ing], subsequences, detected_boxes, overlap_thres, camchange):
        
        # step 1 2 compute group id instead of sequence id
        score_triplet = mlist[0][2]
        if count_mlist < t_step_2:
          dif_num = False
          id1, score1 = compute_group_id(groups[m1ing], subseq_ids_scores, jockeys, track)
          id2, score2 = compute_group_id(groups[m2ing], subseq_ids_scores, jockeys, track)
          if id1 != id2 and id1 != 0 and id2 != 0:
           dif_num = True
          score_scn_1 = 1 / (1 + math.exp(-sigma * score1))
          score_scn_2 = 1 / (1 + math.exp(-sigma * score2))
          if dif_num:
           score_scn_1 = 1 - score_scn_1
           score_scn_2 = 1 - score_scn_2
          score_scn = score_scn_1 * score_scn_2
          score_all = score_triplet * score_scn

          if score_all <= match_thres:
           if score_triplet > triplet_thres * 2:  # because score_triplet = score_cap + score_jockey
             insert_ok = False
             j = t_step_2 - count_mlist
             if score_triplet > mlist[j][2]:
               mlist.insert(j, mlist[0])
               insert_ok = True
             else:
               for i in range(j+1, len(mlist)):
                 if score_triplet <= mlist[i-1][2] and score_triplet >= mlist[i][2]:
                   mlist.insert(i, mlist[0])
                   insert_ok = True
                   break
             if not insert_ok:
               mlist.append(mlist[0])
          
          if score_triplet > triplet_thres * 2:
            insert_ok = False
            j = t_step_2 - count_mlist
            if score_triplet > mlist[j][2]:
              mlist.insert(j, mlist[0])
              insert_ok = True
            else:
              for i in range(j+1, len(mlist)):
                if score_triplet <= mlist[i-1][2] and score_triplet >= mlist[i][2]:
                  mlist.insert(i, mlist[0])
                  insert_ok = True
                  break
            if not insert_ok:
              mlist.append(mlist[0])
        
        if count_mlist >= t_step_2:
          groups[m1ing] = merge_group(groups[m1ing], groups[m2ing], overlap_thres)
          groups.pop(m2ing)
        #if different number but triplet similar, append to the end of matching list
        elif score_triplet > triplet_thres * 2:
            mlist.append(mlist[0])
    
      mlist.pop(0)
      count_mlist += 1
    
    # add back subsequences that are not assigned to any groups
    for i in range(len(subsequences)):
      if i not in matched:
        groups.append([[i, subsequences[i][0,0], subsequences[i][0,-1]]])
    
    #compute id 
    id_groups = []
    for i_g, g in enumerate(groups):
      id_s, score_max = compute_group_id(g, subseq_ids_scores, jockeys, track)
      id_groups.append([id_s,score_max])
    
    # check id duplicate, if two group has the same id, then we need to devide which group to keep its id, other group to have the id remove
    sseq_ids = [sseq_id[0] for sseq_id in id_groups]
    dup = set([x for x in sseq_ids if sseq_ids.count(x) > 1 and x != 0])
    be_merged = []
    while len(dup) > 0:
      dup_i = dup.pop()
      for i,sseq_id in enumerate(id_groups):
        if i in be_merged:
          continue
        for j,sseq_id in enumerate(id_groups):
          if j in be_merged:
            continue
          if i < j and id_groups[i][0] == dup_i and id_groups[j][0] == dup_i:
            if check_overlap_gps(groups[i], groups[j], subsequences, detected_boxes, overlap_thres) and \
               check_jump_gps(groups[i], groups[j], subsequences, detected_boxes, overlap_thres, camchange):
               groups[i]  = merge_group(groups[i], groups[j], overlap_thres)
               id_groups[i][1] += id_groups[j][1]
               be_merged.append(j)
            else:
              # calculate the length of subsequnce i and j
              len_i = sum([subsequences[groups[i][ii][0]].shape[1] for ii in range (0, len(groups[i]))])
              len_j = sum([subsequences[groups[j][jj][0]].shape[1] for jj in range (0, len(groups[j]))])
              if len_i < len_j and len_i < len_thres: #if subsequnce i is shorter than j and i is shorter than len_thres, set its id to 0
                id_groups[i][0] = 0
              elif len_j < len_i and len_j < len_thres:  #if subsequnce j is shorter than i and j is shorter than len_thres, set its id to 0
                id_groups[j][0] = 0
              elif id_groups[i][1] < id_groups[j][1]: # set the subsequnce with lower score to 0
                id_groups[i][0] = 0
              else:
                id_groups[j][0] = 0

    if len(be_merged) > 0:
      groups = [group for i, group in enumerate(groups) if i not in be_merged]
      id_groups = [id_group for i, id_group in enumerate(id_groups) if i not in be_merged]
    
    # compute the cap triplet feature for each group
    group_feats = []
    for g in groups:
      avg_feat = np.zeros((triplet_mean_seq[0].shape[0]))
      frm_no = 0
      for m in g:
        avg_feat += triplet_mean_seq[m[0]] * subsequences[m[0]].shape[1]
        frm_no += subsequences[m[0]].shape[1]
      group_feats.append(avg_feat/frm_no)

    be_merged = []

    #add last connection by nearest triplet cap feature for each group
    # this function was originally using histogram. But now it has switched to triplet features
    ind_new = config.tracking[track].scn_max
    for i,sseq_id in enumerate(id_groups):
      if id_groups[i][0] == 0:
        score_hist_best = 0
        score_hist_second = 0
        selected_j = -1
        length_selected_j = 0
        length_i = 0
        for ii in range(len(groups[i])):
          length_i += subsequences[groups[i][ii][0]].shape[1]
        if length_i < 10:
          continue

        for j,group in enumerate(groups):
          length_j = 0
          for jj in range(len(groups[j])):
            length_j += subsequences[groups[j][jj][0]].shape[1]
          #if length_j < 
          if j != i and id_groups[j][0] != 0 and id_groups[j][0] <= config.tracking[track].scn_max and \
            check_overlap_gps(groups[i], groups[j], subsequences, detected_boxes, 3) and \
            check_jump_gps(groups[i], groups[j], subsequences, detected_boxes, overlap_thres, camchange):   #thres = 3 strict
              score = math.exp(-np.sum((group_feats[i] - group_feats[j]) ** 2) ** 0.5)
              if score > score_hist_best:
                score_hist_second = score_hist_best
                score_hist_best = score
                if score > triplet_thres_gp: # 0.6 0.55
                  selected_j = j
                  length_selected_j = length_j
              elif score > score_hist_second:
                score_hist_second = score

        if selected_j != -1 and score_hist_second / score_hist_best < 0.8 and length_selected_j > len_thres:
          groups[selected_j]  = merge_group(groups[selected_j], groups[i], overlap_thres)
          id_groups[selected_j][1] += id_groups[i][1]
          be_merged.append(i)
          print('merge by group triplet, dist = ', score_hist_best, score_hist_second, len(groups[i]), length_i, length_selected_j)
        else:
          ind_new += 1
          id_groups[i][0] = ind_new

    if len(be_merged) > 0:
      groups = [group for i, group in enumerate(groups) if i not in be_merged]
      id_groups = [id_group for i, id_group in enumerate(id_groups) if i not in be_merged]

    return groups, id_groups




[docs]def check_overlap_gps(g1, g2, sqs, dc, t):
  
  '''
  This function checks whether the two groups of subsequnces can be merged together.
  It will loop over the subsequences in both groups one by one to see if there is any overlapping or switching between them.
  the merging will be rejected if any of these cases happen.
  '''
  
  for m1 in g1:
    for m2 in g2:
      overlap_set = list(set(sqs[int(m1[0])][0,:]).intersection(set(sqs[int(m2[0])][0,:])))
      overlap = list(set(range(int(m1[1]), int(m1[2])+1)).intersection(set(range(int(m2[1]), int(m2[2])+1))))
      overlap_length = len(overlap)
      
      # if the overlap of subsequences is greater than the threshold t, over if the overlap length equals the length of either subsequence, return False, (group should not be joint together)
      if overlap_length > t or len(overlap_set) == sqs[int(m1[0])].shape[1] or len(overlap_set) == sqs[int(m2[0])].shape[1]:
        return False
          
  # if nothing wrong, approve the merge of the two groups
  return True

  

[docs]def merge_group(g1,g2,t):
  
  '''merge the two groups, insert subsequences of group 2 one by one into group 1'''

  for m2 in g2:
    inserted = 0
    for mid, m1 in enumerate(g1):
      if m2[2] < m1[1] + t:
        g1.insert(mid, m2)
        inserted = 1
        break
    if inserted == 0:
      g1.append(m2)
      
  return g1



[docs]def compute_group_id(group, subseq_ids_scores, jockeys, track):
  '''
  compute the jockey id for the group, and the score of the id
  
  '''
  id_gp = 0
  id_score = 0
  lst_score = [0] * config.tracking[track].scn_max
  for sq in group:
    id_tmp = subseq_ids_scores[sq[0]][0]
    score_tmp = subseq_ids_scores[sq[0]][1]
    if id_tmp >= 1:
      id_tmp -= 1
      lst_score[id_tmp] += score_tmp

  lst_score_neg = [-lst for lst in lst_score]
  arg_list = list(np.argsort(np.asarray(lst_score_neg)))
  lst_score = [lst_score[i] for i in arg_list]

  if lst_score[0] > 0:
    id_gp = arg_list[0] + 1
    id_score = lst_score[0] - lst_score[1]
    if id_score == 0:
      id_gp = 0
    if len(jockeys) > 0 and id_gp not in jockeys:
      id_gp = 0
      id_score = 0
  return id_gp, id_score



[docs]def sort_list_by_col(mylist):
  '''sort the list based on the scores of the subsequence pairs'''
  len_list = [m[-1] for m in mylist]
  arg_list = list(np.argsort(np.asarray(len_list)))
  mylist = [mylist[i] for i in arg_list]
  mylist.reverse()
  return mylist



[docs]def check_jump_seqs(a, b, sqs, detected_boxes, overlap_thres):
  
  '''
  check whether the subsequence has jumping too fast in time
  Normally, when the race progress from frame t to frame t+n. The jockey cap will gradrally move from the spatial position at t to the new position at t+n.
  Therefore, if we are connecting the cap between frame t and frame t+n, if n is small, the distance moved of the cap should be small. And such a distance gradually increases, as n becomes larger.
  This function will reject subsequence matching with unexpectedly large spatial displacement of the caps.
  '''
  
  dist = 0.0
  dist_thres = 0.8
  dist_thres_low = 0.3
  a_start_ifrm = sqs[a][0,0]
  a_start_icap = sqs[a][1,0]
  a_end_ifrm = sqs[a][0,sqs[a].shape[1]-1]
  a_end_icap = sqs[a][1,sqs[a].shape[1]-1]
  a_stard_box = detected_boxes[a_start_ifrm][a_start_icap]
  a_end_box = detected_boxes[a_end_ifrm][a_end_icap]

  b_start_ifrm = sqs[b][0,0]
  b_start_icap = sqs[b][1,0]
  b_end_ifrm = sqs[b][0,sqs[b].shape[1]-1]
  b_end_icap = sqs[b][1,sqs[b].shape[1]-1]
  b_stard_box = detected_boxes[b_start_ifrm][b_start_icap]
  b_end_box = detected_boxes[b_end_ifrm][b_end_icap]

  dt = min(abs(a_end_ifrm - b_start_ifrm), abs(b_end_ifrm - a_start_ifrm))
  if dt == 0:
    dt = 0.5
  if dt > 200:
    return 1.0

  if (a_end_ifrm - overlap_thres < b_start_ifrm):
    b1 = a_end_box
    b2 = b_stard_box
  elif (b_end_ifrm - overlap_thres < a_start_ifrm):
    b1 = b_end_box
    b2 = a_stard_box
  else: #overlap
    return 0.0

  dist = (((b1[0] + b1[2]) / 2.0 - (b2[0] + b2[2]) / 2.0) ** 2 + ((b1[1] + b1[3]) / 2.0 - (b2[1] + b2[3]) / 2.0) ** 2 ) ** 0.5
  dist /= ((b1[2] - b1[0]) * (b1[3] - b1[1])) ** 0.5    #box w and h should not be zero
  dist = (dist - 1.0) / dt   # add -1 to make more tolerent to t = 1 2 3
  if (dist > dist_thres):    # dist_thres = 2.0;  sigma ~ 0.15
    return 0.0

  st_score = 1.0
  if dist > dist_thres_low:
    st_score = 1 - 2.0 * (dist - dist_thres_low)

  return st_score

[docs]def check_jump_gps(gp1, gp2, subsequences, detected_boxes, overlap_thres, camchange):
  
  '''
  check whether there is any jumping for subsequence pairs in subrace
  '''
  
  num_subrace = len(camchange) + 1
  subraces = [[] for i in range (0, num_subrace)]
  if len(camchange) == 0:
    subraces[0] = [id for id, sq in enumerate(subsequences)]
  else:
    subraces[0] = [id for id, sq in enumerate(subsequences) if sq[0,-1] <= camchange[0]]
  if num_subrace > 1:
    subraces[-1] = [id for id, sq in enumerate(subsequences) if sq[0,0] > camchange[-1]]
    for i in range(1, num_subrace-1):
      subraces[i] = [id for id, sq in enumerate(subsequences) if sq[0,0] > camchange[i-1] and sq[0,-1] <= camchange[i]]

  for m1 in gp1:
    for m2 in gp2:
      for subrace in subraces:
        if m1[0] in subrace and m2[0] in subrace:
          #only check if two seqs in same subrace
          if check_jump_seqs(m1[0], m2[0], subsequences, detected_boxes, overlap_thres) < 0.001:
            return False
  return True
