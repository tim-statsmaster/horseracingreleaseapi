    def __init__(self, all_imgs:list, all_jk_masks:list, all_boxes:list, all_frame_objs:list, inp_size:int):
        self.all_imgs = all_imgs
        self.all_jk_masks = all_jk_masks
        self.all_boxes = all_boxes
        self.all_frame_objs = all_frame_objs
        self.inp_size = inp_size
        
    def __getitem__(self, idx):
        bgr_frame = decode_img(self.all_imgs[idx])
        frame_objs = self.all_frame_objs[idx]
        jk_masks = self.all_jk_masks[idx]
        bboxes = self.all_boxes[idx]
        
        orig_h, orig_w = bgr_frame.shape[:2]
        rgb_frame = cv2.cvtColor(bgr_frame, cv2.COLOR_BGR2RGB)
        
        imgs = []
        norm_bboxes = []
        data_len = len(bboxes)
        
        if len(bboxes) < 2:  # no matching is needed for number of bboxes < 2
            return imgs, norm_bboxes, data_len       

        for i, box in enumerate(bboxes):
            # =================== Extract object images ===================
            obj_type = frame_objs[i].item()
            y1, x1, y2, x2 = box

            if obj_type == 1:  # Jockey
                img = (rgb_frame[y1:y2, x1:x2, :] * jk_masks[i][..., np.newaxis]).astype(np.uint8)
                img = Image.fromarray(img)
                img = self.preprocess(img)
            else:
                img = rgb_frame[y1:y2, x1:x2, :].astype(np.uint8)
                img = Image.fromarray(img)
                img = self.preprocess(img)
                
            imgs.append(img)
            # =================== Extract bboxes ===================
            x1 = x1/orig_w
            y1 = y1/orig_h
            x2 = x2/orig_w
            y2 = y2/orig_h
            norm_bboxes.append(torch.tensor([x1, y1, x2, y2]))
            
        if len(imgs) > 0:
            imgs = torch.stack(imgs, dim=0)
            norm_bboxes = torch.stack(norm_bboxes, dim=0)
        
        return imgs, norm_bboxes, data_len
    

[docs]    def preprocess(self, img: Image.Image) -> torch.tensor:
        w, h = img.size
        long_edge = max(w, h)
        resize_ratio = self.inp_size / long_edge
        resize_shape = (round(h*resize_ratio), round(w*resize_ratio))
        w_diff, h_diff = (self.inp_size - resize_shape[1]), (self.inp_size - resize_shape[0])
        l_pad = w_diff//2
        r_pad = w_diff - l_pad
        t_pad = h_diff//2
        b_pad = h_diff - t_pad
        padding = (l_pad, t_pad, r_pad, b_pad)

        transform = transforms.Compose([
            transforms.Resize(resize_shape),  # interpolation `BILINEAR` is applied by default
            transforms.Pad(padding=padding, fill=0, padding_mode='constant'),
            transforms.ToTensor(),
            transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])
        ])
        image = transform(img)

        return image

    
    def __len__(self):
        return len(self.all_imgs)
    

[docs]    def collate_fn(self, batch_data):
        imgs, norm_bboxes, data_len = zip(*batch_data)
        imgs = [img for img in imgs if len(img) > 1]
        norm_bboxes = [box for box in norm_bboxes if len(box) > 1]
        data_len = torch.tensor(data_len)
        if len(imgs) == 0:
            return [], [], data_len
        else:
            imgs = torch.cat(imgs, dim=0)
            norm_bboxes = torch.cat(norm_bboxes, dim=0)

        return imgs, norm_bboxes, data_len

    def __init__(self, all_feats:list, all_graphs:list, all_frame_objs:list, k_at_hop:list, active_connection:int):
        self.all_feats = all_feats
        self.all_graphs = all_graphs
        self.all_frame_objs = [frm_objs for frm_objs in all_frame_objs if frm_objs.shape[0] > 1]
        
        self.k_at_hop = k_at_hop
        self.active_connection = active_connection
        self.depth = len(k_at_hop)
        
    def __getitem__(self, idx):
        frm_feats = self.all_feats[idx]
        frm_graph = self.all_graphs[idx]
        frame_objs = self.all_frame_objs[idx]
        
        feat, A, center_idx, one_hop_idcs, unique_nodes_list, obj_mask = [[] for _ in range(6)]

        for node_i in range(frm_feats.shape[0]):
            feat_i, A_i, center_idx_i, one_hop_idcs_i, unique_nodes_list_i, obj_mask_i = self.get_ips_of_node_i(node_i, frm_feats, frm_graph, frame_objs)
            feat.append(feat_i)
            A.append(A_i)
            center_idx.append(center_idx_i)
            one_hop_idcs.append(one_hop_idcs_i)
            unique_nodes_list.append(unique_nodes_list_i)
            obj_mask.append(obj_mask_i)
            
        feat, A, center_idx, one_hop_idcs, obj_mask = map(lambda x: torch.stack(x), 
                                                            [feat, A, center_idx, one_hop_idcs, obj_mask])
        
        return feat, A, center_idx, one_hop_idcs, unique_nodes_list, obj_mask
    
    def __len__(self):
        return len(self.all_feats)
    

[docs]    def get_ips_of_node_i(self, node_i:int, frame_features:torch.tensor, frame_graph:torch.tensor, frame_objs:torch.tensor):
        '''
        Construct Instance-pivot subgraph of node i
        (refer to )
        '''
        # hops[0] for 1-hop neighbors, hops[1] for 2-hop neighbors
        hops = list()
        center_node = node_i
        hops.append(set(frame_graph[center_node][1:].tolist()))

        # Actually we dont need the loop since the depth is fixed here,
        # But we still remain the code for further revision
        for d in range(1,self.depth): 
            hops.append(set())
            for h in hops[-2]:
                hops[-1].update(set(frame_graph[h][1:self.k_at_hop[d]+1].tolist()))

        
        hops_set = set([h for hop in hops for h in hop])
        hops_set.update([center_node,])
        unique_nodes_list = [h for h in hops_set] 
        unique_nodes_map = {j:i for i,j in enumerate(unique_nodes_list)}

        center_idx = torch.Tensor([unique_nodes_map[center_node],]).type(torch.long)
        one_hop_idcs = torch.Tensor([unique_nodes_map[i] for i in hops[0]]).type(torch.long)
        center_feat = frame_features[center_node]
        feat = frame_features[unique_nodes_list]

        feat = feat - center_feat
        
        max_num_nodes = self.k_at_hop[0] * (self.k_at_hop[1] + 1) + 1
        num_nodes = len(unique_nodes_list)
        A = torch.zeros(num_nodes, num_nodes)

        _, fdim = feat.shape
        feat = torch.cat([feat, torch.zeros(max_num_nodes - num_nodes, fdim)], dim=0)  # [max_num_nodes, fdim]
      
        # Create Adjacency Matrix
        for node in unique_nodes_list:
            neighbors = frame_graph[node, 1:self.active_connection+1]
            for n in neighbors:
                n = n.item()
                if n in unique_nodes_list: 
                    A[unique_nodes_map[node], unique_nodes_map[n]] = 1
                    A[unique_nodes_map[n], unique_nodes_map[node]] = 1

        D = A.sum(1, keepdim=True)
        A = A.div(D)
        A_ = torch.zeros(max_num_nodes,max_num_nodes)
        A_[:num_nodes,:num_nodes] = A  # [max_num_nodes, max_num_nodes]
        
        obj_types = frame_objs[torch.tensor(unique_nodes_list)]
        one_hop_obj_types = obj_types[one_hop_idcs]
        center_obj_type = obj_types[center_idx]
        obj_mask = (center_obj_type != one_hop_obj_types).long()
        
        return feat, A_, center_idx, one_hop_idcs, unique_nodes_list, obj_mask

    def __init__(self, model_dir, feat_agg_path, gcn_path, f_dim, k1, k_at_hop, active_connection, inp_size, dist_metric, device, thres=0.5, batch_size=32, n_workers=8, pin_memory=False):

        self.k_at_hop = k_at_hop
        self.active_connection = active_connection
        self.depth = len(k_at_hop)
        self.inp_size = inp_size
        self.thres = thres
        self.device = device
        self.batch_size = batch_size
        self.n_workers = n_workers
        self.pin_memory = pin_memory
        
        self.dist_metric = dist_metric
            
        self.feat_agg = FeatureAggregator(f_dim)
        self.feat_agg.load_state_dict(torch.load(os.path.join(model_dir, feat_agg_path)))
        self.gcn = GCN(in_dim = f_dim)
        self.gcn.load_state_dict(torch.load(os.path.join(model_dir, gcn_path)))
        
        self.feat_agg.to(device)
        self.gcn.to(device)
        
        self.feat_agg.eval()
        self.gcn.eval()
    

[docs]    def predict(self, all_imgs:list, all_masks:list, all_bboxes:list, all_frame_objs:list):
        
        all_feats, all_graphs, obj_cnts = self.get_feats_and_graph(all_imgs, all_masks, all_bboxes, all_frame_objs)
        
        graph_set = GraphDataset(all_feats, all_graphs, all_frame_objs, self.k_at_hop, self.active_connection)
        graph_loader = DataLoader(graph_set, batch_size=self.batch_size, shuffle=False, num_workers=self.n_workers, collate_fn=graph_set.collate_fn, pin_memory=self.pin_memory)
        
        all_preds = []
        all_h1ids = []
        all_obj_masks = []
        all_unique_nodes_lists = []
        # for bi, (feat, A, center_idx, one_hop_idcs, unique_nodes_list, obj_mask) in enumerate(tqdm(graph_loader, desc="GCN matching: ")):
        for bi, (feat, A, center_idx, one_hop_idcs, unique_nodes_list, obj_mask) in enumerate(graph_loader):

            for fi, Ai, ci, oi, ui, mi in zip(feat, A, center_idx, one_hop_idcs, unique_nodes_list, obj_mask):

                fi, Ai, ci, oi, mi = map(lambda x: x.to(self.device), 
                                                [fi, Ai, ci, oi, mi])

                with torch.no_grad():
                    pred = self.gcn(fi, Ai, oi)
                pred = F.softmax(pred, dim=1)
                pred = pred.view(fi.shape[0], oi.shape[1], 2)
            
                all_preds.append(pred)
                all_h1ids.append(oi)
                all_obj_masks.append(mi)
            all_unique_nodes_lists += unique_nodes_list
        
                
        for i, n_nodes in enumerate(obj_cnts):
            if n_nodes < 2:
                all_preds.insert(i, -1)
                all_obj_masks.insert(i, -1)
                all_h1ids.insert(i, -1)
                all_unique_nodes_lists.insert(i, -1)
                
        all_clusters = []
        all_cluster_confs = []
        for frm_i, n_nodes in enumerate(obj_cnts):
            if n_nodes > 1:
                pred = all_preds[frm_i]
                obj_mask = all_obj_masks[frm_i]
                h1id = all_h1ids[frm_i]
                frame_objs = all_frame_objs[frm_i]
                unique_nodes_list = all_unique_nodes_lists[frm_i]
                clusters, cluster_confs = self.extract_clusters_from_matrix(pred, obj_mask, frame_objs, h1id, unique_nodes_list)
            else:
                clusters, cluster_confs = [], []
                
            all_clusters.append(clusters)
            all_cluster_confs.append(cluster_confs)
            
    
        return all_clusters, all_cluster_confs

    

[docs]    def get_feats_and_graph(self, all_imgs:list, all_masks:list, all_bboxes:list, all_frame_objs:list) -> (list, list):
        '''
        Get features and up to k1 nearest neighbors
        Args:
        - batch_bgr_frame: list of np.arrays (each array belongs to a single frame)
        - batch_masks: list of np.arrays (each array belongs to a single frame)
        - batch_bboxes: list of list (each nested list belongs to a single frame)
        - batch_frame_objs: list of tensors (each tensor belongs to a single frame)
        
        Returns:
        - feats: list of tensors (each tensor belongs to a single frame)
        - graphs: list of tensors (each tensor belongs to a single frame)
        '''
        
        feat_set = FeatureDataset(all_imgs, all_masks, all_bboxes, all_frame_objs, inp_size=self.inp_size)
        feat_loader = DataLoader(feat_set, batch_size=self.batch_size, shuffle=False, num_workers=self.n_workers, collate_fn=feat_set.collate_fn, pin_memory=self.pin_memory)
        
        graphs = []
        feats = []
        obj_cnts = []
        # for bi, (imgs, norm_bboxes, lens) in enumerate(tqdm(feat_loader, desc="Processing features for matching: ")):
        for bi, (imgs, norm_bboxes, lens) in enumerate(feat_loader):

            obj_cnts += [n.item() for n in lens]
            
            if len(imgs) < 1:
                continue
            
            imgs, norm_bboxes = map(lambda x: x.to(self.device), [imgs, norm_bboxes])
        
            with torch.no_grad():
                frm_feats = self.feat_agg(imgs, norm_bboxes).cpu()

            res_i = 0
            for n_nodes in lens:
                n_nodes = n_nodes.item()
                if n_nodes > 1:
                    c_start = sum(lens[:res_i]) if res_i != 0 else 0
                    candidates = frm_feats[c_start:c_start+n_nodes]
                    a = candidates.unsqueeze(1)  # [n_nodes, 1, f_dim]
                    b = candidates.repeat((n_nodes,1,1))  # [n_nodes, n_nodes, f_dim]
                    if self.dist_metric == 'cosine':
                        node_dists = 1 - nn.CosineSimilarity(dim=-1, eps=1e-6)(a, b)  # [n_nodes, n_nodes]
                    elif self.dist_metric == 'l2':
                        node_dists = torch.cdist(a, b, p=2).squeeze(1)  # [n_nodes, n_nodes]

                    sorted_neighbors = node_dists.squeeze(-1).sort(dim=-1, descending=False).indices  # [n_nodes, n_nodes]
                    top_k1_nn = sorted_neighbors[:, :self.k1+1]  # [n_nodes, k1+1]
                    graphs.append(top_k1_nn)
                    feats.append(candidates)
                    res_i += 1
                
        return feats, graphs, obj_cnts

    

[docs]    def extract_clusters_from_matrix(self, pred:torch.tensor, obj_mask:torch.tensor, frame_objs:torch.tensor, one_hop_idcs:torch.tensor, unique_nodes_list:list):
        '''
        Args:
        - pred (torch.int64): [n_nodes, k1]
        '''
        
        pred_confs = (pred[:, :, 1] * obj_mask.float()).cpu() * (pred[:, :, 1] > self.thres).float().cpu()

        a = torch.zeros(pred_confs.shape[0], pred_confs.shape[0])
        for r in range(pred_confs.shape[0]):
            n_real_ids = torch.tensor(unique_nodes_list[r])[one_hop_idcs[r].cpu()].long()
            a[r, n_real_ids] = pred_confs[r]

        pairs = []
        clus_mat = torch.zeros(pred_confs.shape[0], pred_confs.shape[0])
        while a.sum().item() != 0:
            r, c = list(zip(*torch.where(a == a.max())))[0]
            conf = a.max().item()
            # Zero out
            a[r, c] = 0
            a[c, r] = 0
            r_type_ids = torch.where(frame_objs == frame_objs[r])[0]
            c_type_ids = torch.where(frame_objs == frame_objs[c])[0]
            a[r, c_type_ids] = 0  # For r-th node, it cannot match with other c-type nodes
            a[c, r_type_ids] = 0  # For c-th node, it cannot match with other r-type nodes
            a[c_type_ids, r] = 0  # For every c-type node, it cannot match with r-th node
            a[r_type_ids, c] = 0  # For every r-type node, it cannot match with c-th node
            if clus_mat[r, :].sum().item() < 3 and clus_mat[c, :].sum().item() < 3:
                clus_mat[r, c] = 1
                clus_mat[c, r] = 1
                pairs.append((r.item(), c.item(), conf))

        # Clean subsets
        clusters = []
        cluster_confs = []
        nodes2clus = {}
        visited_nodes = set()
        pairs = sorted(pairs, key=lambda x: x[-1], reverse=True)  # sorted by confs
        for (n1, n2, conf) in pairs:
            if nodes2clus.get(n1) is None and nodes2clus.get(n2) is None:
                clus_id = len(clusters)
                nodes2clus[n1] = clus_id
                nodes2clus[n2] = clus_id
                clusters.append([n1, n2])
                cluster_confs.append([conf])
            elif nodes2clus.get(n1) is not None and nodes2clus.get(n2) is None:
                clus_id = nodes2clus[n1]
                if len(clusters[clus_id]) < 3:
                    nodes2clus[n2] = clus_id
                    clusters[clus_id].append(n2)
                    cluster_confs[clus_id].append(conf)
            elif nodes2clus.get(n1) is None and nodes2clus.get(n2) is not None:
                clus_id = nodes2clus[n2]
                if len(clusters[clus_id]) < 3:
                    nodes2clus[n1] = clus_id
                    clusters[clus_id].append(n1)
                    cluster_confs[clus_id].append(conf)
            elif nodes2clus.get(n1) is not None and nodes2clus.get(n2) is not None:  # merge 2 clusters
                n1_clus_id = nodes2clus[n1]
                n2_clus_id = nodes2clus[n2]
                if n1_clus_id == n2_clus_id:
                    cluster_confs[n1_clus_id].append(conf)
                    continue
                if len(clusters[n1_clus_id] + clusters[n2_clus_id]) <= 3:
                    nodes2clus[n1] = n1_clus_id
                    for ni in clusters[n2_clus_id]:
                        nodes2clus[ni] = n1_clus_id
                    clusters[n1_clus_id] += clusters[n2_clus_id]
                    cluster_confs[n1_clus_id] = cluster_confs[n1_clus_id] + cluster_confs[n2_clus_id]
                    clusters.pop(n2_clus_id)
                    cluster_confs.pop(n2_clus_id)

        clusters = [set(i) for i in clusters]
        cluster_confs = [np.mean(conf) for _, conf in sorted(zip(clusters, cluster_confs), key=lambda x: max(x[0]))]
        clusters = sorted(clusters, key=lambda x: max(x))
        
        # TODO: also return singular cluster (or discard?)

        return clusters, cluster_confs
