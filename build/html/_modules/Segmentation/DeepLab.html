import tensorflow
import tensorflow.compat.v1 as tf
tf.disable_v2_behavior()

USE_FUSED_BN = True
BN_EPSILON = 0.001
BN_MOMENTUM = 0.99


[docs]def relu_separable_bn_block(inputs, filters, dilation_rate, name_prefix, is_training, reuse, data_format, relu=True):
    bn_axis = -1 if data_format == 'channels_last' else 1

    if relu==True:
        inputs = tf.nn.relu(inputs, name=name_prefix + '_act')

    inputs = tf.layers.separable_conv2d(inputs, filters, (3, 3),
                        strides=(1, 1), padding='same', dilation_rate=dilation_rate,
                        data_format=data_format,
                        activation=None, use_bias=False,
                        depthwise_initializer=tensorflow.initializers.GlorotUniform(),
                        pointwise_initializer=tensorflow.initializers.GlorotUniform(),
                        bias_initializer=tf.zeros_initializer(),
                        name=name_prefix, reuse=reuse,  trainable=is_training)
    inputs = tf.layers.batch_normalization(inputs, momentum=BN_MOMENTUM, name=name_prefix + '_bn', axis=bn_axis,
                            epsilon=BN_EPSILON, training=is_training, reuse=reuse, fused=USE_FUSED_BN)

    if relu != True:
        inputs = tf.nn.relu(inputs, name=name_prefix + '_act')

    return inputs

[docs]def XceptionModel(input_image, is_training = False, reuse=False, data_format='channels_last'):
    bn_axis = -1
    # Entry Flow
    inputs = tf.layers.conv2d(input_image, 32, (3, 3), use_bias=False, name='block1_conv1', strides=(2, 2),
                padding='valid', data_format=data_format, activation=None, reuse=reuse, trainable=is_training,

                kernel_initializer=tensorflow.initializers.GlorotUniform(),
                bias_initializer=tf.zeros_initializer())
    inputs = tf.layers.batch_normalization(inputs, momentum=BN_MOMENTUM, name='block1_conv1_bn', axis=bn_axis,
                            epsilon=BN_EPSILON, training=is_training, reuse=reuse, fused=USE_FUSED_BN)
    inputs = tf.nn.relu(inputs, name='block1_conv1_act')

    inputs = tf.layers.conv2d(inputs, 64, (3, 3), use_bias=False, name='block1_conv2', strides=(1, 1),
                padding='valid', data_format=data_format, activation=None, reuse=reuse, trainable=is_training,
                kernel_initializer=tensorflow.initializers.GlorotUniform(),
                bias_initializer=tf.zeros_initializer())
    inputs = tf.layers.batch_normalization(inputs, momentum=BN_MOMENTUM, name='block1_conv2_bn', axis=bn_axis,
                            epsilon=BN_EPSILON, training=is_training, reuse=reuse, fused=USE_FUSED_BN)
    inputs = tf.nn.relu(inputs, name='block1_conv2_act')

    residual = tf.layers.conv2d(inputs, 128, (1, 1), use_bias=False, name='conv2d_1', strides=(2, 2),
                padding='same', data_format=data_format, activation=None, reuse=reuse, trainable=is_training,
                kernel_initializer=tensorflow.initializers.GlorotUniform(),
                bias_initializer=tf.zeros_initializer())
    residual = tf.layers.batch_normalization(residual, momentum=BN_MOMENTUM, name='batch_normalization_1', axis=bn_axis,
                            epsilon=BN_EPSILON, training=is_training, reuse=reuse, fused=USE_FUSED_BN)

    inputs = tf.layers.separable_conv2d(inputs, 128, (3, 3),
                        strides=(1, 1), padding='same',
                        data_format=data_format,
                        activation=None, use_bias=False,
                        depthwise_initializer=tensorflow.initializers.GlorotUniform(),
                        pointwise_initializer=tensorflow.initializers.GlorotUniform(),
                        bias_initializer=tf.zeros_initializer(),
                        name='block2_sepconv1', reuse=reuse,  trainable=is_training)
    inputs = tf.layers.batch_normalization(inputs, momentum=BN_MOMENTUM, name='block2_sepconv1_bn', axis=bn_axis,
                            epsilon=BN_EPSILON, training=is_training, reuse=reuse, fused=USE_FUSED_BN)

    inputs = relu_separable_bn_block(inputs, 128, 1, 'block2_sepconv2', is_training, reuse, data_format)

    stride_2_features = inputs

    inputs = tf.layers.max_pooling2d(inputs, pool_size=(3, 3), strides=(2, 2),
                                    padding='same', data_format=data_format,
                                    name='block2_pool')

    inputs = tf.add(inputs, residual, name='residual_add_0')
    residual = tf.layers.conv2d(inputs, 256, (1, 1), use_bias=False, name='conv2d_2', strides=(2, 2),
                padding='same', data_format=data_format, activation=None, reuse=reuse,  trainable=is_training,
                kernel_initializer=tensorflow.initializers.GlorotUniform(),
                bias_initializer=tf.zeros_initializer())
    residual = tf.layers.batch_normalization(residual, momentum=BN_MOMENTUM, name='batch_normalization_2', axis=bn_axis,
                            epsilon=BN_EPSILON, training=is_training, reuse=reuse, fused=USE_FUSED_BN)

    inputs = relu_separable_bn_block(inputs, 256, 1, 'block3_sepconv1', is_training, reuse, data_format)
    inputs = relu_separable_bn_block(inputs, 256, 1, 'block3_sepconv2', is_training, reuse, data_format)

    stride_4_features = inputs

    inputs = tf.layers.max_pooling2d(inputs, pool_size=(3, 3), strides=(2, 2),
                                    padding='same', data_format=data_format,
                                    name='block3_pool')
    inputs = tf.add(inputs, residual, name='residual_add_1')

    residual = tf.layers.conv2d(inputs, 728, (1, 1), use_bias=False, name='conv2d_3', strides=(2, 2),
                padding='same', data_format=data_format, activation=None, reuse=reuse,  trainable=is_training,
                kernel_initializer=tensorflow.initializers.GlorotUniform(),
                bias_initializer=tf.zeros_initializer())
    residual = tf.layers.batch_normalization(residual, momentum=BN_MOMENTUM, name='batch_normalization_3', axis=bn_axis,
                            epsilon=BN_EPSILON, training=is_training, reuse=reuse, fused=USE_FUSED_BN)

    inputs = relu_separable_bn_block(inputs, 728, 1, 'block4_sepconv1', is_training, reuse, data_format)
    inputs = relu_separable_bn_block(inputs, 728, 1, 'block4_sepconv2', is_training, reuse, data_format)

    stride_8_features = inputs

    inputs = tf.layers.max_pooling2d(inputs, pool_size=(3, 3), strides=(2, 2),
                                    padding='same', data_format=data_format,
                                    name='block4_pool')
    inputs = tf.add(inputs, residual, name='residual_add_2')
    # Middle Flow
    for index in range(8):
        residual = inputs
        prefix = 'block' + str(index + 5)

        inputs = relu_separable_bn_block(inputs, 728, 1, prefix + '_sepconv1', is_training, reuse, data_format)
        inputs = relu_separable_bn_block(inputs, 728, 1, prefix + '_sepconv2', is_training, reuse, data_format)
        inputs = relu_separable_bn_block(inputs, 728, 1, prefix + '_sepconv3', is_training, reuse, data_format)
        inputs = tf.add(inputs, residual, name=prefix + '_residual_add')

    Xception_outputs = inputs

    return Xception_outputs, stride_2_features, stride_4_features, stride_8_features

    

[docs]def decoder(ASPP_outputs, low_level_features, concat_feature_size=48, final_feature_size=256,is_training = False, reuse= False, name="1"):

    data_format='channels_last'

    height, width = low_level_features.shape[1:3]
    upsampled_outputs = tf.image.resize_bilinear(ASPP_outputs, [height, width], align_corners=True)

    projected_outputs = tf.layers.conv2d(low_level_features, concat_feature_size, (1, 1), use_bias=True, name='Project_Output_' + name, strides=(1, 1),
                     padding='same', activation=tf.nn.relu, reuse=reuse,
                     kernel_initializer=tensorflow.initializers.GlorotUniform(),
                     bias_initializer=tf.zeros_initializer())

    projected_outputs = tf.layers.batch_normalization(projected_outputs, momentum=BN_MOMENTUM, name='Project_Output_bn_' + name, axis=-1,
                            epsilon=BN_EPSILON, training=is_training, reuse=reuse, fused=USE_FUSED_BN)

    concat_outputs = tf.concat([upsampled_outputs, projected_outputs], 3)

    decoder_features = relu_separable_bn_block(concat_outputs, final_feature_size, 1, 'decoder_' + name + '_conv0', is_training, reuse, data_format, False)
    decoder_features = relu_separable_bn_block(decoder_features, final_feature_size, 1, 'decoder_' + name + '_conv1', is_training, reuse, data_format, False)

    return decoder_features

    

[docs]def deeplab(input_image, n_class, is_train=False, reuse=True):

    Xception_outputs, stride_2_features, stride_4_features, stride_8_features = XceptionModel(input_image, is_training = is_train, reuse = reuse, data_format='channels_last')
    decoder_outputs = decoder(Xception_outputs, stride_8_features, concat_feature_size=96, final_feature_size=256, is_training = is_train, reuse = reuse, name="1")
    decoder_outputs = decoder(decoder_outputs, stride_4_features, concat_feature_size=48, final_feature_size=128, is_training = is_train, reuse = reuse, name="2")
    decoder_outputs = decoder(decoder_outputs, stride_2_features, concat_feature_size=32, final_feature_size=64, is_training = is_train, reuse = reuse, name="3")

    height, width = input_image.shape[1:3]
    upsampled_image = tf.image.resize_bilinear(decoder_outputs, [height, width], align_corners=True)

    output_map = tf.layers.conv2d(upsampled_image, n_class, (1, 1), use_bias=False, name='Final_Output', strides=(1, 1),
                     padding='same', activation=None, reuse=reuse,
                     kernel_initializer=tensorflow.initializers.GlorotUniform(),
                     bias_initializer=tf.zeros_initializer())

    return output_map
