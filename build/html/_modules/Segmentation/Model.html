import numpy as np

import tensorflow.compat.v1 as tf

from Segmentation.DeepLab import deeplab


[docs]class Model(object):
    """
    Class for using DeepLab Model
    
    Parameters
    ----------
    network_path : string
    	path for the network to restore
    channels : int
    	input channels, default is 3
    n_class : int
    	number of output class
    img_row : int
    	width of input image
    img_cols : int
    	height of input image
    cropping : int
    	the amount of cropping at test, optimum is 35
    """

    def __init__(self, network_path, channels=3, n_class=1, img_rows=112, img_cols=112, cropping=0):

        '''initialise the model and load the saved weights'''
        
        tf.reset_default_graph()

        self.x_shape = tf.placeholder("float", shape=[None, img_rows, img_cols, channels], name="x")
        self.y_shape = tf.placeholder("float", shape=[None, img_rows - 2 * cropping, img_cols - 2 * cropping, n_class], name="y")

        self.model_test = deeplab(self.x_shape, n_class, is_train=False, reuse=False)
        logits_test = self.model_test[:,cropping:img_rows-cropping,cropping:img_cols-cropping,:]
        
        if n_class == 1:
            self.pred_test = tf.round(tf.sigmoid(logits_test))
        else:
            self.pred_test = tf.argmax(tf.nn.softmax(logits_test), -1)

        gpu_config = tf.ConfigProto()
        gpu_config.gpu_options.allow_growth = True
        self.sess = tf.Session(config=gpu_config)
        self.restore(network_path)
                        

[docs]    def predict(self, x_test):
        """
        Uses the model to create a prediction for the given data
        
        Parameters
        ----------
        x_test :  
        	Data to predict on. Shape [n, nx, ny, channels]
        
        Returns
        -------
        pred : numpy array
        	Prediction has only 2 dimensions, width and height. each element is the predicted class number on that particular pixel
        """
        
        x_test = np.expand_dims(x_test,0) if len(x_test.shape) == 3 else x_test
        pred = np.reshape(self.sess.run(self.pred_test, feed_dict={self.x_shape: x_test}), (self.y_shape.shape[1] ,self.y_shape.shape[2]))
        
        return pred
