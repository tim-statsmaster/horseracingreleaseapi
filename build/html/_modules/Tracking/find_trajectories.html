[docs]def get_dist_change_input(cb1, cb2):
  ''' 
  The dist change as input for the connectivity network
  
  '''
  cx1, cy1 = (cb1[2] + cb1[0]) / 2.0, (cb1[3] + cb1[1]) / 2.0 # Center of cap1
  cx2, cy2 = (cb2[2] + cb2[0]) / 2.0, (cb2[3] + cb2[1]) / 2.0 # Center of cap2
  cb2_size = ( cb2[2] - cb2[0] ) * ( cb2[3] - cb2[1] ) # Cap 2 width * height
  return abs( cx2 - cx1 ) / ( (cb2_size ** 0.5) * 100 ), abs( cy2 - cy1 ) / ( (cb2_size ** 0.5) * 50 )

[docs]def triplet_feature_to_list(feat_npy, detected_boxes):
  '''
  convert the structure of the triplet feature into the same as detected_boxes, i.e. list of list of features
  
  Parameters
  ----------
  feat_npy : numpy array
  	triplet feature in the shape of (total item number, feature_vector_size). If the detected caps are 10000, then, shape is (10000, 128) If the detected jockeys are 8000, then shape is (8000, 2048).
  	Note the difference in feature vector size, as the caps triplet model is still the old one. 
  detected_boxes: list of list of list
  	for every frame, for every cap/jockey, there is a bounding box
  Returns
  -------
  feat_list : list of list of vector(numpy array)
    	the triplet feature arranged in similar format as detected_boxes
  '''
  feat_list = []
  count = 0
  for frm in range(len(detected_boxes)):
    feat_list.append([])
    for cbid in range(len(detected_boxes[frm])):
      feat_list[-1].append(feat_npy[count,:])
      count += 1
  return feat_list

[docs]def get_trajectory(horseracingResult):

    '''
    the main function of the tracking algorithm.
    Tracking proceduce
    1) Load the data obtained in the previous tasks
    2) Predict the saddlecloth numbers from the saddlecloth images
    3) Compute the cap features from the cap images
    4) Compute the jockey features from the jockey images
    5) Compute the cap connectivity score from the feature distance calculated from the cap features, the weigthed pixel displacements in the x and y direction, and the frame difference.
    	Discard the unwanted connections based on global min cost criteria and the connectivity threshold
    6) Join the caps at frame difference == 1 based on the connectivity score to get the subsequences
    7) break the subsequences based on the connectivity scores and triplet features
    8) Initial Tracking: Join the subsequences at frame differences > 1 and < 5 based on the connectivity scores and triplet features. Ignore subsequences pairs from different subrace
    9) Obtain the probability of a certain subsequence belongs to a particular jockey number, using the result from 2)
    10) Final Tracking: Join the subsequences across subraces (at any point in the race) with the cap features, jockey features, jockey number prediction for the subsequences, known jockey numbers for the race
    11) Clean the subsequences based on the known jockey numbers, and output the final result
    '''

    # 1) Load the data obtained in the previous tasks
    # get the race information
    model_dir = horseracingResult.model_dir
    result_dir = horseracingResult.result_dir
    vid_name = horseracingResult.processing_vid_name

    print('get trajectory ... ')
    track = horseracingResult.race_info[vid_name]["track"]
    distance = horseracingResult.race_info[vid_name]["distance"]
    course = horseracingResult.race_info[vid_name]["course"]
    
    jockeys = horseracingResult.jockeys[vid_name] if vid_name in horseracingResult.jockeys.keys() else []
    print("jockeys" , jockeys)
    
    start_frame, end_frame = horseracingResult.start_end_frames[vid_name]
    camchange = horseracingResult.cam_changes[vid_name]

    print('Starting Frame is', start_frame, 'and Ending Frame is', end_frame)
    print("Camera change Frames: ", camchange)

    if start_frame >= end_frame or start_frame < 0 or end_frame < 0:
      print('Start/End frame error, end of tracking')
      return
    
    # Read the information from the json file obtained from detection
    detected_cap_boxes, detected_jockey_boxes, detected_jockey_masks, detected_sc_boxes, cap_jockey_relation, stf, edf = read_detection_results(result_dir, config.directories[config.Task.DETECTION]['ROOT'], vid_name, config.tracking)

    # accept if only one frame difference, to delete after testing
    if abs(stf-start_frame) == 1:
      start_frame = stf
      horseracingResult.start_end_frames[vid_name] = start_frame, end_frame 
    if abs(edf-end_frame) == 1:
      end_frame = edf
      horseracingResult.start_end_frames[vid_name] = start_frame, end_frame 

    assert(stf==start_frame and edf==end_frame)

    #camchange = [c-start_frame for c in camchange]
    camchange = [[int(c-start_frame) for c in cc] for cc in camchange]
    print("Camera change Frames: ", camchange)
    #delete boxes in camera changing frames
    for cc in camchange:
      num_cc_frames = cc[1] - cc[0] + 1
      detected_cap_boxes[cc[0]:cc[1]+1] = [[] for i in range(num_cc_frames)]
      detected_jockey_boxes[cc[0]:cc[1]+1] = [[] for i in range(num_cc_frames)]
      detected_jockey_masks[cc[0]:cc[1]+1] = [[] for i in range(num_cc_frames)]
      detected_sc_boxes[cc[0]:cc[1]+1] = [[] for i in range(num_cc_frames)]
      cap_jockey_relation[cc[0]:cc[1]+1] = [[] for i in range(num_cc_frames)]

    print("Cap Boxes: ", len([ff for f in range(len(detected_cap_boxes)) for ff in detected_cap_boxes[f]]))
    print("Jockey Boxes: ", len([ff for f in range(len(detected_jockey_boxes)) for ff in detected_jockey_boxes[f]]))
    print("Saddlecloth Boxes: ", len([ff for f in range(len(detected_sc_boxes)) for ff in detected_sc_boxes[f]]))

    #2) Predict the saddlecloth numbers from the saddlecloth images
    # get the saddlecloth scores ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
    fi = glob.glob(os.path.join(result_dir, config.directories[config.Task.TRACKING]['SADDLECLOTHNUMBER'], vid_name + ".npy"))
    if len(fi) > 0: # load saved results if true
      detected_sc_boxes_save = np.load(fi[0], 'r', True)
      detected_sc_boxes = [[] for i in range(len(detected_sc_boxes))]
      for i in range(detected_sc_boxes_save.shape[0]):
        detected_sc_boxes[detected_sc_boxes_save[i,0]].append(detected_sc_boxes_save[i,1:].tolist())
    else:
      horseracingResult.load_frames(vid_name, start_frame, end_frame)
      saddlecloths = []
      for frm in range(len(detected_sc_boxes)):
        frame = decode_img(horseracingResult.frames[frm])
        for s in detected_sc_boxes[frm]:
          saddlecloths.append(box2im(s, frame))

      # saddlecloths = [box2im(s, decode_img(horseracingResult.frames[frm])) for frm in range(len(detected_sc_boxes)) for s in detected_sc_boxes[frm]]
      # detect saddlecloth id
      detected_sc_boxes = get_ids_scores(saddlecloths, detected_sc_boxes, track, model_dir, jockeys)
      detected_sc_boxes_save = np.asarray([ [f] + box for f in range(len(detected_sc_boxes)) for box in detected_sc_boxes[f]])

      if b_save_cache:
        np.save(os.path.join(result_dir, config.directories[config.Task.TRACKING]['SADDLECLOTHNUMBER'], vid_name +".npy"), detected_sc_boxes_save, True)
    
    # 3) Compute the cap features from the cap images
    # Obtain Triplet Cap Features----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
    fi = glob.glob(os.path.join(result_dir, config.directories[config.Task.TRACKING]['CAPTRIPLET'], vid_name + "*.npy"))
    if len(fi) > 0:
      tii = time.time()
      triplet_features_caps_npy = np.load(fi[0], "r", True)
      triplet_features_caps = triplet_feature_to_list(triplet_features_caps_npy, detected_cap_boxes)
      print("TripFeats loaded: " + str(time.time()-tii))
    else:
      # get triplet embedding for all jockey caps ----------------------------------------------------------------------------------------------------
      ttfs = time.time()
      horseracingResult.load_frames(vid_name, start_frame, end_frame)
      cap_images = []
      for frm in range(len(detected_cap_boxes)):
        frame = decode_img(horseracingResult.frames[frm])
        for cbid in range(len(detected_cap_boxes[frm])):
          cap_image = box2im(detected_cap_boxes[frm][cbid], frame)
          cap_images.append(cap_image)
      triplet_features_caps_npy = copy.deepcopy(get_feature(model_dir, track, cap_images))
      print("Get Cap Triplet Features in time: " + str(time.time()-ttfs))
      
      if b_save_cache:
        np.save(os.path.join(result_dir, config.directories[config.Task.TRACKING]['CAPTRIPLET'], vid_name + ".npy"), triplet_features_caps_npy, True)
      triplet_features_caps = triplet_feature_to_list(triplet_features_caps_npy, detected_cap_boxes)

    #4) Compute the jockey features from the jockey images
    # Obtain Triplet Jockey Features------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
    ji = glob.glob(os.path.join(result_dir, config.directories[config.Task.TRACKING]['JOCKEYTRIPLET'], vid_name + "*.npy"))
    if len(ji) > 0:
      tii = time.time()
      triplet_features_jockeys_npy = np.load(ji[0], "r", True)
      triplet_features_jockeys = triplet_feature_to_list(triplet_features_jockeys_npy, detected_jockey_boxes)
      print("TripFeats loaded: " + str(time.time()-tii))
    else:
      # get triplet embedding for all jockey masks ----------------------------------------------------------------------------------------------------
      ttfs = time.time()
      horseracingResult.load_frames(vid_name, start_frame, end_frame)
      jockey_images = []
      triplet_features_jockeys_npy = get_jockey_feature(model_dir, track, horseracingResult.frames, detected_jockey_boxes, detected_jockey_masks)
      print("Get Jockey Triplet Features in time: " + str(time.time()-ttfs))
      
      if b_save_cache:
        np.save(os.path.join(result_dir, config.directories[config.Task.TRACKING]['JOCKEYTRIPLET'], vid_name + ".npy"), triplet_features_jockeys_npy, True)
      triplet_features_jockeys = triplet_feature_to_list(triplet_features_jockeys_npy, detected_jockey_boxes)

    ### Get GCN matching score 
    # get matching result ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
    cap_sc_match_scores = cap_saddlecloth_matching(detected_cap_boxes, detected_sc_boxes, detected_jockey_boxes, detected_jockey_masks, horseracingResult)
    
    # 5) Compute the cap connectivity score from the feature distance calculated from the cap features, the weigthed pixel displacements in the x and y direction, and the frame difference.
    #    Discard the unwanted connections based on global min cost criteria and the connectivity threshold
    # get connectivity scores using the cap triplet features ------------------------------------------------------------------------------------------
    fi = glob.glob(os.path.join(result_dir, config.directories[config.Task.TRACKING]['CONNECTIVITY'], vid_name + "*"))
    if len(fi) > 0:
      tii = time.time()
      selected_data = np.load(fi[0], "r", True)
      delim = np.asarray([ int(d) for d in fi[0][:-4].split("&")[1:]])
      selected_results = [selected_data[:,np.sum(delim[:d]):np.sum(delim[:d+1])] for d in range(delim.shape[0])]
      print([sr.shape for sr in selected_results])
      print("Conn feature loaded: " + str(time.time()-tii))
    else:
      tttt = time.time()
               
      all_results = []
      for t in [1,2,3,4]: # compute the connectivity scores for frame differences 1,2,3 and 4
        current_results = np.array([]).reshape(7,0)
        #print('len(detected_cap_boxes) = ', len(detected_cap_boxes))  #2716
        total_pairs = []
        for frm in range(len(detected_cap_boxes) - t):

          caps1_boxes = detected_cap_boxes[frm]
          caps2_boxes = detected_cap_boxes[frm+t]
          num_pairs = len(caps1_boxes) * len(caps2_boxes) # the possible cap pairs across frames

          ### DEBUG
          total_pairs += [num_pairs]

          # create numpy array to store the following data ( f1, f2, cap1_id, cap2_id, dx, dy, score) * num_pairs
          result = np.zeros((7,num_pairs))
          result[0,:] = frm
          result[1,:] = frm + t

          t1 = time.time()
          count = 0
          for c1 in range(len(caps1_boxes)):
            for c2 in range(len(caps2_boxes)):
              t2 = time.time()
              dx, dy = get_dist_change_input(caps1_boxes[c1], caps2_boxes[c2])
              j1 = cap_jockey_relation[frm][c1] # get the jockey body that relate the the cap frm, c1
              j2 = cap_jockey_relation[frm+t][c2] # get the jockey body that relate the the cap frm, c2
              if j1 != -1 and j2 != -1: 
                # compute the distance between the pair of jockey body features if the jockey is not -1 (jockey == -1 if there is no matching jockey body for that particular cap)
                jdist = np.sum((triplet_features_jockeys[frm][j1] - triplet_features_jockeys[frm+t][j2]) ** 2) ** 0.5
              else:
                jdist = -1
              # 0.4 if the threshold here. j_adjust is a weight of adjustment depending on jdist. if jdist > 0.4, j_adjust > 1, and the cap distance will be bigger.
              # this measure fuses the cap feature info with the jockey feature info to make the prediction more robust  
              j_adjust = (jdist/0.4) ** 0.5 if jdist != -1 else 1

              dist = np.sum((triplet_features_caps[frm][c1] - triplet_features_caps[frm+t][c2]) ** 2) ** 0.5 * j_adjust
              result[2:,count] = [c1, c2, dx, dy, dist]
              count += 1
  
          # here count == num_pairs
          # result is for a pair of frames, current_results concatenate all frames
          current_results = np.concatenate([current_results, result], 1)
        
        #all frames
        print("t={},total_pairs={},avg_per_frame={},total_frames={}".format(t,sum(total_pairs),sum(total_pairs)/len(total_pairs),len(detected_cap_boxes)))
        
        # get costs
        # Get connectivity cost
        data = np.zeros((current_results.shape[1], 4), dtype=np.float32) 
        data[:,0] = current_results[-1,:] # dist 
        data[:,1] = current_results[-3,:] # dx
        data[:,2] = current_results[-2,:] # dy
        data[:,3] = t/2.0 # frame difference divided by 2
        #replace triplet score with connectivity cost
        current_results[-1,:] = get_costs(model_dir, track, distance, course, data)
        # current_results all frames, all_results all t = 1 2 3 4 of all frames
        all_results.append(current_results)
      
      print([ar.shape  for ar in all_results])
      # remove pairs with cost higher than the connectivity threshold
      print('conn_thres =', config.tracking[track].conn_thres)
      reduced_results = [al[:, al[6,:] < config.tracking[track].conn_thres] for al in all_results]
      print([ar.shape  for ar in reduced_results])
  
      # Reduce duplicated pairs (1 cap connected to 2 or more) by selecting global min cost
      selected_results = []
       # Sort from min cost
      reduced_results = [rr[:,rr[-1,:].argsort()] for rr in reduced_results]
      for t in range(1):
        ind = 0
        dic = set() # keep track on the selected cap connections, skip if the connection is already selected
        current_results = []
        while ind < reduced_results[t].shape[1]: 
          f1, f2, c1, c2 = reduced_results[t][:4,ind]
          if str(int(f1)) + "-" + str(int(c1))  + "-" in dic or "-" + str(int(f2)) + "-" + str(int(c2)) in dic:
            ind += 1
            continue
          current_results.append(copy.deepcopy(reduced_results[t][:,ind]))
          dic.add(str(int(f1)) + "-" + str(int(c1)) + "-")
          dic.add("-" + str(int(f2)) + "-" + str(int(c2)))
          ind += 1
        selected_current_results = np.transpose(np.asarray(current_results), (1,0))

        selected_current_results = selected_current_results[:,selected_current_results[0,:].argsort()]
        selected_results.append(selected_current_results)
    
      selected_results += reduced_results[1:]
    
      print([ar.shape  for ar in selected_results])

      # save selected results to numpy file
      res_len = "&".join([str(sr.shape[1]) for sr in selected_results])
      selected_data = np.concatenate(selected_results, 1)
      if b_save_cache:
        np.save(os.path.join(result_dir, config.directories[config.Task.TRACKING]['CONNECTIVITY'], vid_name + "&" + res_len + ".npy"), selected_data, True)
      print("Conn Feature Saved: " + str(time.time()-tttt))
  
    # Obtain subsequences, break at camchange forward (connect at t=1 only) -----------------------------------------------------------------------------
    si = glob.glob(os.path.join(result_dir, config.directories[config.Task.TRACKING]['SUBSEQUENCE'], vid_name + ".npy"))
    tii = time.time()
    if len(si) > 0:
      subsequences_data = np.load(si[0], "r", True)
      f = open(os.path.join(result_dir, config.directories[config.Task.TRACKING]['SUBSEQUENCE'], vid_name + ".txt"))
      res_len = f.read()
      delim = np.asarray([ int(d) for d in res_len.split("&")])
      subsequences = [subsequences_data[:,np.sum(delim[:d]):np.sum(delim[:d+1])] for d in range(delim.shape[0])]
      print("Subsequences loaded: " + str(time.time()-tii))
      f.close()
    else:
      # connect the caps into subsequences using the connectivity score at frame difference == 1
      subsequences = link_subsequences(selected_results)
      # Sort the subsequences by the length of each subsequences
      subsequences = sort_subsequences(subsequences)
      
      # save selected results to numpy file
      res_len = "&".join([str(ss.shape[1]) for ss in subsequences])
      subsequences_data = np.concatenate(subsequences, 1)
      if b_save_cache:
        np.save(os.path.join(result_dir, config.directories[config.Task.TRACKING]['SUBSEQUENCE'], vid_name + ".npy"), subsequences_data, True)
        f = open(os.path.join(result_dir, config.directories[config.Task.TRACKING]['SUBSEQUENCE'], vid_name + ".txt"), "w")
        f.write(res_len)
        f.close()
      print("Subsequences Saved: " + str(time.time()-tii))
    
    print('sum caps:', sum([sq.shape[1] for sq in subsequences]))
    print('(After link subsequence) num of subsequences:', len(subsequences))

  
    # ----------------------------------------------------------------------------------------------------------------------------------------------------
    # break_trip_thres init_trip_thres are related to distance, not similarity
    break_intervals, break_trip_thres, init_intervals, init_trip_thres = [3], [1.1,0.8,0.7], [2,3,4], [0.5,0.3]

    # break subsequences
    subsequences = break_subsequences(subsequences, selected_results, triplet_features_caps, triplet_features_jockeys, cap_jockey_relation, break_intervals, break_trip_thres)

    #extract middle frame of camchange
    camchange = [int((c[0]+c[1])*0.5) for c in camchange]

    print('(After break subsequence) num of subsequences:', len(subsequences))

    subseq_ids_scores = update_subsequences_scn(subsequences, cap_sc_match_scores, detected_cap_boxes, detected_sc_boxes, horseracingResult)
    
    # sort subsequences by jockey id probablity
    subsequences, subseq_ids_scores = sort_subsequences4(subsequences, subseq_ids_scores)
    
    # initial tracking
    subsequences = initial_tracking(subsequences, selected_results, triplet_features_caps, triplet_features_jockeys, cap_jockey_relation, camchange, init_intervals, track, init_trip_thres, subseq_ids_scores)

    print('(After initial_tracking) num of subsequences:', len(subsequences))
    # remove short sequences
    #subsequences = remove_short_subsequences(subsequences, config.tracking[track].remove_thres)
    remove_thres = int(len(subsequences)/500)
    if remove_thres > 0:
      subsequences = remove_short_subsequences(subsequences, remove_thres)
      print('(After remove short) num of subsequences:', len(subsequences))

    subsequences = sort_subsequences(subsequences)
    print('sum caps:', sum([sq.shape[1] for sq in subsequences]))
    print('num of initial subsequences:', len(subsequences))
    
    # link the saddlecloth to the caps, and compute a score for each subsequences on the associated jockey number
    subseq_ids_scores = update_subsequences_scn(subsequences, cap_sc_match_scores, detected_cap_boxes, detected_sc_boxes, horseracingResult)

    # sort subsequences by jockey id probablity
    subsequences, subseq_ids_scores = sort_subsequences4(subsequences, subseq_ids_scores)

    # final matching
    subsequences, subsequences_ids = match_with_triplet_saddlecloth(subsequences, subseq_ids_scores, triplet_features_caps, triplet_features_jockeys, cap_jockey_relation, detected_cap_boxes, camchange, jockeys, track)
   
    # if only one unknown number, assign id from sectional time file (run only when there are input jockeys)
    if len(jockeys) > 0:
      subsequences, subsequences_ids = clean_subsequence_with_jockey_numbers(subsequences, subsequences_ids, jockeys, track)
    
    print([s.shape[1] for s in subsequences])
    print('sum caps:', sum([sq.shape[1] for sq in subsequences]))

    # export results to csv
    export_results_with_id(os.path.join(result_dir, config.directories[config.Task.TRACKING]['FINALRESULT']), vid_name, 1080, 1920, subsequences, subsequences_ids, detected_cap_boxes, start_frame)
  
    # for debug use
    if b_visualise:
      if not os.path.exists(result_dir + "/visualise/"):
        os.makedirs(result_dir + "/visualise/tracking/")
      if not os.path.exists(result_dir + "/visualise/tracking/"):
        os.makedirs(result_dir + "/visualise/tracking/")
      path = result_dir + "/visualise/tracking/" + vid_name + "/"
      if not os.path.exists(path):
        os.makedirs(path)
      if horseracingResult.frames is None:
        horseracingResult.load_frames(vid_name, start_frame, end_frame)
      visualise(path, horseracingResult.frames, subsequences, subsequences_ids, detected_cap_boxes, start_frame)

    gc.collect()

    return
