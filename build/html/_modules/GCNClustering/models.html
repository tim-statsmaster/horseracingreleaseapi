[docs]class GraphConv(nn.Module):
    def __init__(self, in_dim, out_dim, agg):
        super().__init__()
        self.in_dim = in_dim
        self.out_dim = out_dim
        self.weight = nn.Parameter(
                torch.FloatTensor(in_dim *2, out_dim))
        self.bias = nn.Parameter(torch.FloatTensor(out_dim))
        init.xavier_uniform_(self.weight)
        init.constant_(self.bias, 0)
        self.agg = agg()


[docs]    def forward(self, features, A):
        b, n, d = features.shape
        assert(d==self.in_dim)
        agg_feats = self.agg(features,A)
        cat_feats = torch.cat([features, agg_feats], dim=2)
        out = torch.einsum('bnd,df->bnf', (cat_feats, self.weight))
        out = F.relu(out + self.bias)
        return out 

        


[docs]class GCN(nn.Module):
    def __init__(self, in_dim):
        super().__init__()
        self.bn0 = nn.BatchNorm1d(in_dim, affine=False)
        self.conv1 = GraphConv(in_dim, in_dim, MeanAggregator)
        self.conv2 = GraphConv(in_dim, in_dim, MeanAggregator)
        self.conv3 = GraphConv(in_dim, in_dim//2, MeanAggregator)
        self.conv4 = GraphConv(in_dim//2, in_dim//4,MeanAggregator)
        
        self.classifier = nn.Sequential(
                            nn.Linear(in_dim//4, in_dim//4),
                            nn.PReLU(in_dim//4),
                            nn.Linear(in_dim//4, 2))
    

[docs]    def forward(self, x, A, one_hop_idcs, train=True):
        # data normalization l2 -> bn
        B,N,D = x.shape
        #xnorm = x.norm(2,2,keepdim=True) + 1e-8
        #xnorm = xnorm.expand_as(x)
        #x = x.div(xnorm)
        
        x = x.view(-1, D)
        x = self.bn0(x)
        x = x.view(B,N,D)

        x = self.conv1(x,A)
        x = self.conv2(x,A)
        x = self.conv3(x,A)
        x = self.conv4(x,A)
        k1 = one_hop_idcs.size(-1)
        dout = x.size(-1)
        edge_feat = torch.zeros(B,k1,dout).cuda()
        for b in range(B):
            edge_feat[b,:,:] = x[b, one_hop_idcs[b]]  
        edge_feat = edge_feat.view(-1,dout)
        pred = self.classifier(edge_feat)
            
        # shape: (B*k1)x2
        return pred



[docs]class FeatureAggregator(nn.Module):
    def __init__(self, feat_dim):
        super().__init__()
        self.img_encoder =  models.resnet18(pretrained=False)
        self.img_encoder.eval()
        self.img_head = self.create_hidden_layer(1000, 512)
        self.bbox_head = self.create_hidden_layer(4, 512)
        self.agg_head = self.create_hidden_layer(512, 1024)
        self.final_layer = nn.Linear(1024, feat_dim)
    

[docs]    def forward(self, img, bbox):
        '''
        img: (bs, 3, h, w)
        bbox: (bs, 4)
        '''
        img_feat = self.img_encoder(img)
        img_feat = self.img_head(img_feat)
        bbox_feat = self.bbox_head(bbox)
#         feat = torch.cat([img_feat, bbox_feat], dim=-1)
        feat = img_feat + bbox_feat
        feat = self.agg_head(feat)
        feat = self.final_layer(feat)
        return feat

    

[docs]    def create_hidden_layer(self, in_dim, out_dim):
        layer = nn.Sequential(
            nn.Linear(in_dim, out_dim),
            nn.BatchNorm1d(out_dim),
            nn.LeakyReLU(0.01)
        )
        return layer
