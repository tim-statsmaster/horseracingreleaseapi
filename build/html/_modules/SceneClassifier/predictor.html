    def _setup(self):
        """
            Setup the parameters at initialisation

        """
        ## mean and std of the model (ImageNet mean and std)
        self.mean = np.array([0.485, 0.456, 0.406])
        self.std = np.array([0.229, 0.224, 0.225])

        self.resolution = (336, 336)
        # prestart, postend, fade, running
        # classes_id_map = {  "prestart":0,
        #                     "postend":1,
        #                     "sudden":2,
        #                     "fade":3,
        #                     "running":4
        #                 }
        self.classes_id_map = {  "prestart":0,
                            "postend":1,
                            "fade":2,
                            "running":3
                        }
        self.num_cls = len(self.classes_id_map)

        ## window size (number of frames) of prediction interval
        self.window_size = 21
        ## threshold for start, end, sudden, fade
        self.start_threshold, self.end_threshold,self.fade_threshold = (0.8, 0.8,0.59)
        #start_threshold, end_threshold, sudden_threshold,fade_threshold = (.8000, .9500, .4286,.8367)
        self.fade_interval_max_threshold = .8
        #sudden_interval_max_threshold = .8

        ## speed stats cal from race records, will be used to predict end frame
        ## Added 3% buffer for wrong start frame detection
        self.max_speed = 18.3*1.03
        self.min_speed = 14.5*0.97

        ## max start interval in sec between positive start prediction
        self.max_start_interval = 2.2

        ## max distance (in sec) between continues fade predictions
        self.max_fade_distance = 1
        ## max distance (in sec) between continues sudden predictions
        #max_sudden_distance = 0.5

        ## number of perd per frame
        self.num_pred_per_frame = 3
        ## a prediction will be run if #pred_interval frames are updated
        self.pred_interval = self.window_size//self.num_pred_per_frame

    def _load_model(self, encoder_path, decoder_path):
        """
            Load the Scene Classification model

        Args:
            encoder_path (string): Path to SceneClassification Encoder model
            decoder_path (string): Path to SceneClassification Encoder model
        """
        ### Load Classification Model
        self.device = torch.device("cuda:0" if torch.cuda.is_available() else "cpu")
        print('using: ', self.device)
        encoder = ResNetEncoder(resnet_deepth=50,
                        pretrain=True,
                        fc_hidden1=512,
                        fc_hidden2=128,
                        dropout_prob=0.3,
                        output_dim=128)

        decoder = LSTMDecoder(input_dim=128,
                            num_lstm_layers=2,
                            lstm_hidden=128,
                            fc_dim=56,
                            dropout_prob=0.3,
                            num_classes=self.num_cls)
        self.encoder = encoder.to(self.device)
        self.decoder = decoder.to(self.device)
        self.encoder.load_state_dict(torch.load(encoder_path))
        self.decoder.load_state_dict(torch.load(decoder_path))

        self.encoder.eval()
        self.decoder.eval()

[docs]    def predict(self, video_path, distance):
        cap = cv2.VideoCapture(video_path)
        self.cap_fps = round(cap.get(cv2.CAP_PROP_FPS))
        ## the model_fps setting must follow the sampling method defined in sampling.py
        if self.cap_fps in (50, 60):
            self.model_fps = 10
        elif self.cap_fps == 25:
            self.model_fps = 12.5

        self.step = int(self.cap_fps/self.model_fps)
        self.max_frame = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))

        ### Adjust the max start time to 40s if the video is shorter than the max_start_time
        if self.max_frame/self.cap_fps<= self.max_start_time:
            self.max_start_time = 40

        ## max/min_run_length in number of frames
        self.min_run_length = int((distance/self.max_speed)*self.cap_fps)
        self.max_run_length = int((distance/self.min_speed)*self.cap_fps)

        ## Read through the video until start,end,camera change are detected/video has ended
        while True:
            success, frame = cap.read()
            if not success:
                break
            self.current_video_idx += 1

            if len(self.frames_indices_lst) == self.window_size:
                self.frames_indices_lst.pop(0)
            ## frames_indices_lst should be [(frame_num1, frame1), (frame_num2, frame2), ...]
            self.frames_indices_lst.append((self.current_video_idx, frame))
            self.used_frame_count += 1

            ## predict only when frames_indices_lst is fulled and enough new frames updated into the list
            if (len(self.frames_indices_lst) == self.window_size) and \
               (self.used_frame_count >= self.pred_interval):
                self.used_frame_count = 0
                #self._predict() # Deprecated
                self._batch_predict()

            ## burn out frames
            for _ in range(self.step-1):
                success, frame = cap.read()
                self.current_video_idx += 1

            ## self._predict() will set self.done to be true if the prediction is done (i.e. start/end/fade all predicted)
            if self.done:
                cap.release()
                break

        all_preds, start_frame, end_frame, cc_lst, cap_fps = self.frames_preds_lst, self.start_frame, self.end_frame, self.cc_lst, self.cap_fps

        return all_preds, start_frame, end_frame, cc_lst, cap_fps


    """
    ### Batch predict, only predict and postprocess if stored batch == batch size
        or the video frames left <= step*window_size*batch_size
    """
    def _batch_predict(self,batch_size = 16):
        """
            Batch predict, only predict and postprocess if stored batch == batch size
            or the video frames left <= step*window_size*batch_size
        Args:
            batch_size (int, optional): number of batches to process at once, default is 16
        """
        indices = [e[0] for e in self.frames_indices_lst]
        frames = self.preprocess(self.frames_indices_lst)

        self.batch_run += [{"indices_lst":indices,"preprocessed_frames":frames}]
        ### Prevent case where video ends early before batch size is matched
        batch_frm_threshold = self.step* self.window_size *batch_size
        vid_frame_left = self.max_frame - self.current_video_idx
        if len(self.batch_run) == batch_size or vid_frame_left<=batch_frm_threshold:
            batched_frames = []
            for each_run in self.batch_run:
                batched_frames+=[each_run["preprocessed_frames"]]

            batched_frames = torch.stack(batched_frames).to(self.device)
            with torch.no_grad():
                X_embedding = self.encoder(batched_frames)
                y_hat = self.decoder(X_embedding).cpu()
                y_hat = y_hat.reshape(-1,self.window_size, self.num_cls)
                y_hat = torch.nn.Softmax(dim=-1)(y_hat)

                for i,each_run in enumerate(self.batch_run):
                    indices = each_run["indices_lst"]
                    self.frames_preds_lst.extend(list(zip(indices, y_hat.numpy()[i])))

            self.batch_run = []
            self._postprocess()


[docs]    def preprocess(self, frames_indices_lst, use_cache = True):
        frames = []
        for indice,frame in frames_indices_lst:
            if use_cache:
                if indice in self.cached_frame:
                    frames += [self.cached_frame[indice]]
                    continue

            frame = cv2.resize(frame, self.resolution, interpolation=cv2.INTER_AREA)
            frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            frame = (frame/255-self.mean)/self.std
            frame = np.moveaxis(frame, -1, 0)
            frames +=[frame]

            if use_cache:
                self.cached_frame[indice] = frame
        frames = np.array(frames)
        frames = frames.astype('float32')
        frames = torch.tensor(frames)

        ### Clean the not used cached frames
        if use_cache:
            min_indice = min([e[0] for e in self.frames_indices_lst])
            for ind in range(min_indice):
                if ind in self.cached_frame:
                    del self.cached_frame[ind]

        return frames


    def _get_avg_pred(self):
        """
            Each frame is being detected 3 times, except for first (self.pred_interval * self.step)*(self.num_pred_per_frame-1) frame.
            This function is called when detect start/end, to compute the average prediction for each frame.
        """
        frame_pred_dict = {}
        for each in self.frames_preds_lst.copy():
            framenum = each[0]
            preds = np.array([each[1]])
            if framenum not in frame_pred_dict:
                frame_pred_dict[framenum] = preds
            else:
                frame_pred_dict[framenum] = np.append(frame_pred_dict[framenum],preds,axis = 0)

        avg_frame_pred_list = []
        for framenum,frame_preds in frame_pred_dict.items():
            ### Consider the frame as valid only when self.num_pred_per_frame is satisfied
            ### except the first (self.pred_interval * self.step -1)*(self.num_pred_per_frame-1) frames
            ### This is to make sure every frame is being predicted for self.num_pred_per_frame times before used
            if framenum >= (self.pred_interval * self.step)*(self.num_pred_per_frame-1):
                if frame_preds.shape[0]<self.num_pred_per_frame:
                    continue

            avg_frame_pred_list += [(framenum,np.mean(frame_preds,axis = 0))]
        return avg_frame_pred_list

    def _predict_start(self, threshold):
        ### Use average prediction for start
        self.frames_avg_preds_lst = self._get_avg_pred()
        ## use the last start prediction of the longest continued start frame prediction interval
        start_id = self.classes_id_map["prestart"]
        start_preds = [idx for idx, pred in self.frames_avg_preds_lst if pred[start_id] >= threshold]
        if len(start_preds) == 0:
            return
        max_continue_start = 0
        carrying_continue_start = 0
        start_frame = start_preds[0]
        for i in range(len(start_preds)-1):
            if (start_preds[i+1] - start_frame) > self.min_run_length:
                break
            if (start_preds[i+1] - start_preds[i])/self.cap_fps <= self.max_start_interval:
                carrying_continue_start += 1
                if carrying_continue_start > max_continue_start:
                    max_continue_start = carrying_continue_start
                    start_frame = start_preds[i+1]
                else:
                    carrying_continue_start = 0
            else:
                break

        self.start_frame = int(start_frame)

    def _predict_end(self, threshold):
        ## use the first end prediction within the possible end frame range calculated from start and distance
        if self.start_frame == -1:
            return

        ### Use average prediction for end
        self.frames_avg_preds_lst = self._get_avg_pred()
        end_id = self.classes_id_map["postend"]
        end_preds = [idx for idx, pred in self.frames_avg_preds_lst if pred[end_id] >= threshold]
        end_preds = [idx for idx in end_preds if idx in range(self.start_frame+self.min_run_length, self.start_frame+self.max_run_length)]
        if len(end_preds) == 0:
            return
        self.end_frame = int(end_preds[0])

    def _predict_cc(self):
        """
            Predict the camera change after start frame and end frame is found
        """
        ## find the fades
        fade_id = self.classes_id_map["fade"]
        print(f'{self.current_video_idx}: finding cc')
        cc_preds = [(idx, pred[fade_id])for idx, pred in self.frames_preds_lst if (pred[fade_id] >= self.fade_threshold and idx in range(self.start_frame, self.end_frame))]
        cc_lst = []
        cc_conf = []
        cc_max_conf = []
        cc_count = []
        for idx, conf in cc_preds:
            included = False
            for i, cc in enumerate(cc_lst):
                ## if current prediction is actually inside the range of one of the existing predictions
                if idx in range(cc[0]-self.cap_fps*self.max_fade_distance, cc[1]+self.cap_fps*self.max_fade_distance):
                    included = True
                    cc_lst[i][0] = min(cc_lst[i][0], idx)
                    cc_lst[i][1] = max(cc_lst[i][1], idx)
                    cc_conf[i] = (cc_conf[i]*cc_count[i]+conf)/(cc_count[i]+1)
                    cc_max_conf[i] = max(cc_max_conf[i], conf)
                    cc_count[i] += 1
                    break
            ## append a new fade range
            if not included:
                cc_conf.append(conf)
                cc_max_conf.append(conf)
                cc_count.append(1)
                cc_lst.append([idx, idx])
        print(f'cc before filtering: {cc_lst}')
        cc_lst = [cc for i, cc in enumerate(cc_lst) if cc_max_conf[i]>self.fade_interval_max_threshold]

        ### Only used with sudden
        if "sudden" in self.classes_id_map:
            ## find the sudden
            sudden_id = self.classes_id_map["sudden"]
            sudden_cc_preds = [(idx, pred[sudden_id])for idx, pred in self.frames_preds_lst if (pred[sudden_id] >= self.sudden_threshold and idx in range(self.start_frame, self.end_frame))]
            sudden_cc_lst = []
            sudden_cc_conf = []
            sudden_cc_max_conf = []
            sudden_cc_count = []
            for idx, conf in sudden_cc_preds:
                included = False
                for i, cc in enumerate(sudden_cc_lst):
                    ## if current prediction is actually inside the range of one of the existing predictions
                    if idx in range(cc[0]-int(self.cap_fps*self.max_sudden_distance), cc[1]+int(self.cap_fps*self.max_sudden_distance)):
                        included = True
                        sudden_cc_lst[i][0] = min(sudden_cc_lst[i][0], idx)
                        sudden_cc_lst[i][1] = max(sudden_cc_lst[i][1], idx)
                        sudden_cc_conf[i] = (sudden_cc_conf[i]*sudden_cc_count[i]+conf)/(sudden_cc_count[i]+1)
                        sudden_cc_max_conf[i] = max(sudden_cc_max_conf[i], conf)
                        sudden_cc_count[i] += 1
                        break
                ## append a new sudden range
                if not included:
                    sudden_cc_conf.append(conf)
                    sudden_cc_max_conf.append(conf)
                    sudden_cc_count.append(1)
                    sudden_cc_lst.append([idx, idx])

            print(f'sudden cc before filtering: {sudden_cc_lst}')
            sudden_cc_lst = [[cc[0]-self.step,cc[1]] for i, cc in enumerate(sudden_cc_lst) if sudden_cc_max_conf[i]>self.sudden_interval_max_threshold]

            ### If its not faded, use sudden_cc_list instead, Assuming theres no mixed sudden and fade cc
            if not len(cc_lst)>0:
                cc_lst = sudden_cc_lst
                cc_conf = sudden_cc_conf
                cc_count = sudden_cc_count
                cc_max_conf = sudden_cc_max_conf

        self.cc_lst = cc_lst
        self.done = True
        print(f'{self.current_video_idx}: found cc = {cc_lst} | avg. conf: {cc_conf} | vote: {cc_count} | max. conf: {cc_max_conf}')

    def _postprocess(self):

        ## find the start_frame, when current time is larger than max_start_time
        ## keep reduce the threshold if start frame can't be found
        if self.start_frame == -1 and (self.current_video_idx/self.cap_fps > self.max_start_time or self.current_video_idx>=self.max_frame-10):
            threshold_discount = 0.85
            factor = 0
            while self.start_frame == -1:
                threshold = self.start_threshold * pow(threshold_discount, factor)
                print(f'{self.current_video_idx}: finding start with threshold: {threshold}')
                self._predict_start(threshold)
                factor += 1
                if self.start_frame != -1:
                    print(f'{self.current_video_idx}: found start = {self.start_frame}')
                ### Added threshold < 0.001: to avoid stuck in this loop
                if threshold < 0.001:
                    print("Start frame cannot be found, Skipping...")
                    break

        ## find the end_frame, when current time is larger than max_run_length plus start_frame
        ## keep reduce the threshold if end frame can't be found
        if self.start_frame != -1 and (self.current_video_idx > self.start_frame+self.max_run_length or self.current_video_idx>=self.max_frame-10):
            threshold_discount = 0.85
            factor = 0
            while self.end_frame == -1:
                threshold = self.end_threshold * pow(threshold_discount, factor)
                print(f'{self.current_video_idx}: finding end with threshold: {threshold}')
                self._predict_end(threshold)
                factor += 1
                if self.end_frame != -1:
                    print(f'{self.current_video_idx}: found end = {self.end_frame}')

                if threshold < 0.001:
                    print("End frame cannot be found, Skipping...")
                    break
