[docs]class ResNetEncoder(nn.Module):
    """
    input shape = (batch_size, timestep, h, w, 3)
    output shape = (batch_size, timestep, output_dim)
    """
    def __init__(self, resnet_deepth=34, pretrain=True, fc_hidden1=512, fc_hidden2=256, dropout_prob=0.3, output_dim=128):
        super().__init__()
        assert resnet_deepth in [18, 34, 50, 101, 152]

        self.fc_hidden1, self.fc_hidden2 = fc_hidden1, fc_hidden2
        self.dropout_prob = dropout_prob

        # resnet = torch.hub.load('pytorch/vision:v0.10.0', f'resnet{resnet_deepth}', pretrained=pretrain)
        if resnet_deepth == 18:
            resnet = torchvision.models.resnet18(pretrained=pretrain, progress=True)
            print("loaded resnet18")
        if resnet_deepth == 34:
            resnet = torchvision.models.resnet34(pretrained=pretrain, progress=True)
            print("loaded resnet34")
        if resnet_deepth == 50:
            resnet = torchvision.models.resnet50(pretrained=pretrain, progress=True)
            print("loaded resnet50")
        if resnet_deepth == 101:
            resnet = torchvision.models.resnet101(pretrained=pretrain, progress=True)
            print("loaded resnet101")
        if resnet_deepth == 152:
            resnet = torchvision.models.resnet151(pretrained=pretrain, progress=True)
            print("loaded resnet152")

        modules = list(resnet.children())[:-1]      # delete the last fc layer.
        self.resnet = nn.Sequential(*modules)
        self.fc1 = nn.Linear(resnet.fc.in_features, fc_hidden1)
        self.bn1 = nn.BatchNorm1d(fc_hidden1, momentum=0.01)
        self.fc2 = nn.Linear(fc_hidden1, fc_hidden2)
        self.bn2 = nn.BatchNorm1d(fc_hidden2, momentum=0.01)
        self.fc3 = nn.Linear(fc_hidden2, output_dim)


[docs]    def forward(self, x_3d):
        cnn_embed_seq = []
        for t in range(x_3d.size(1)):
            with torch.no_grad():
                x = self.resnet(x_3d[:, t, :, :, :])  # ResNet
                x = x.view(x.size(0), -1)             # flatten output of conv

            x = self.bn1(self.fc1(x))
            x = F.relu(x)
            x = self.bn2(self.fc2(x))
            x = F.relu(x)
            x = F.dropout(x, p=self.dropout_prob, training=self.training)
            x = self.fc3(x)

            cnn_embed_seq.append(x)

        # swap time and sample dim such that (sample dim, time dim, CNN latent dim)
        # torch.stack(cnn_embed_seq, dim=0): shape=(time_step, batch, input_size)
        cnn_embed_seq = torch.stack(cnn_embed_seq, dim=0).transpose_(0, 1)
        # cnn_embed_seq: shape=(batch, time_step, input_size)

        return cnn_embed_seq



[docs]class LSTMDecoder(nn.Module):
    """
    input shape = (batch_size, timestep, input_dim)
    output shape = (batch_size, timestep, num_classes)
    """
    def __init__(self, input_dim=128,
                 num_lstm_layers=2,
                 lstm_hidden=128,
                 fc_dim=56,
                 dropout_prob=0.3,
                 num_classes=5):
        super().__init__()

        self.input_dim = input_dim
        self.num_lstm_layers = num_lstm_layers         # RNN hidden layers
        self.lstm_hidden = lstm_hidden                 # RNN hidden nodes
        self.fc_dim = fc_dim
        self.dropout_prob = dropout_prob
        self.num_classes = num_classes

        self.LSTM = nn.LSTM(
            input_size=self.input_dim,
            hidden_size=lstm_hidden,
            num_layers=num_lstm_layers,
            batch_first=True,
            bidirectional=True
        )

        self.fc1 = nn.Linear(self.lstm_hidden*2, self.fc_dim)
        self.fc2 = nn.Linear(self.fc_dim, self.num_classes)


[docs]    def forward(self, x):
        output, (_, _) = self.LSTM(x)
        # FC layers
        output = self.fc1(output)   # choose RNN_out at the last time step
        output = F.relu(output)
        output = F.dropout(output, p=self.dropout_prob, training=self.training)
        output = self.fc2(output)
        # output = torch.sigmoid(output)
        # output = torch.unsqueeze(output, -1)
        return output



[docs]def softmax(x):
    exp_x = torch.exp(x)
    sum_x = torch.sum(exp_x, dim=1, keepdim=True)

    return exp_x/sum_x


[docs]def log_softmax(x):
    return torch.log(softmax(x))



[docs]def CrossEntropyLoss(outputs, targets, weights):
    outputs = log_softmax(outputs)
    outputs = outputs * weights
    # targets = torch.nn.functional.one_hot(targets, num_classes=-1).float()
    weighted_batch_size = (weights * targets).sum()
    outputs = targets * outputs
    return - torch.sum(outputs)/weighted_batch_size
