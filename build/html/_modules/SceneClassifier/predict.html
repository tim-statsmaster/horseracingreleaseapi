[docs]def find_start_end_cc(horseracingResult):
    """

    Loads scene classificiation models and predict start/end frame + Camera Change from it

    Args:
        horseracingResult (object): Storing intermediate Results
    """
    ### Setup variables
    vid_name = horseracingResult.processing_vid_name
    track = horseracingResult.race_info[vid_name]["track"]
    distance = horseracingResult.race_info[vid_name]["distance"]
    course = horseracingResult.race_info[vid_name]["course"]
    target_fps = horseracingResult.get_fps(vid_name) ## 25/30

    ## Find the target video path
    video_path = []
    ### If a path is given in horseracingResult.vid_dir, use it directly
    if not os.path.isdir(horseracingResult.vid_dir):
        video_path = horseracingResult.vid_dir
    ### Otherwise, find all the videos in that folder
    else:
        for format_ in config.accepted_formats:
            video_path += glob.glob(horseracingResult.vid_dir + f'/*{format_}')
        video_path = [p for p in video_path if vid_name in p]
        assert len(video_path) == 1
        video_path = video_path[0]

    ### Load SceneClassification Model and predict
    encoder_path = horseracingResult.model_dir + '/' + config.get_model_paths("SceneClassify", '')['encoder']
    decoder_path = horseracingResult.model_dir + '/' + config.get_model_paths("SceneClassify", '')['decoder']
    predictor = SceneClassificationPredictor(encoder_path, decoder_path)
    
    with predictor:
        _, start_frame, end_frame, cam_change_list, cap_fps = predictor.predict(video_path, distance)

        ## Convert the frame number of the result back to video original fps
        ratio = target_fps/cap_fps
        start_frame = int(np.floor(start_frame*ratio))
        end_frame = int(np.ceil(end_frame*ratio))
        cam_change_list = [[int(np.floor(cc[0]*ratio)),int(np.ceil(cc[1]*ratio))] for cc in cam_change_list]
        print(f'**After converting fps: start: {start_frame} | end: {end_frame} | cc: {cam_change_list}')

        ## Set the prediction result (Start/End,Camera Changes) to Result instance
        horseracingResult.start_end_frames[horseracingResult.processing_vid_name] = [start_frame, end_frame]
        horseracingResult.cam_changes[horseracingResult.processing_vid_name] = cam_change_list

        ### Save Results to vid_process.csv and camera_changes.csv to disk
        ### Save to video_processor.csv
        ### Header = raceLabel,track,course,raceDist,md5,fps,coursePred,raceNumPred,raceDistPred,raceTimePred,startFrmNum,finishFrmNum
        with open(os.path.join(horseracingResult.result_dir, config.video_processer_save_path), "a+") as video_preprocess_file:
            vp_result_row = ",".join([vid_name, track, str(course), str(distance), "", str(target_fps),"","","","",str(start_frame), str(end_frame)])
            video_preprocess_file.write(vp_result_row + "\n")

        ### Save to camera_changes.csv
        ### Header = raceLabel,cc0_start,cc0_end,cc1_start,cc1_end,cc2_start,cc2_end,cc3_start,cc3_end,...
        cam_change_result_row = vid_name
        for cc in cam_change_list:
            cam_change_result_row += "," + str(cc[0])
            cam_change_result_row += "," + str(cc[1])

        with open(os.path.join(horseracingResult.result_dir, config.camerachange_save_path), "a+") as cam_change_file:
            cam_change_file.write(cam_change_result_row + "\n")
